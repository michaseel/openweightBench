    <style>

    <!-- Columns Container -->
    <main id="kanban-board" class="grid grid-cols-1 md:grid-cols-4 gap-6 max-w-7xl mx-auto">

        <div class="bg-white p-6 rounded-lg shadow-2xl max-w-sm w-full transform transition-all scale-100">
            <h3 class="text-xl font-bold text-red-600 mb-4">Confirm Deletion</h3>
            <p class="mb-6 text-gray-700">Are you sure you want to delete this card? This action cannot be undone.</p>
            <!-- The confirm button must use the specific data-testid -->
            <div class="flex justify-end space-x-3">
                <button id="cancel-delete" class="px-4 py-2 bg-gray-200 text-gray-800 rounded hover:bg-gray-300 transition duration-150">Cancel</button>

    <script>
        // ================================================
        // ⚙️ CONFIGURATION AND INITIAL STATE SETUP
        // ================================================

        const STORAGE_KEY = 'kanbanBoardStateV1';
        const COLUMNS = ['backlog', 'in-progress', 'review', 'done'];
        let state = {}; // Holds the entire Kanban board data structure

        // Dummy cards to ensure initial content if localStorage is empty
        const DUMMY_CARDS = [
            { id: 1, title: "Setup initial project architecture", column: "backlog" },
            { id: 2, title: "Implement drag-and-drop functionality", column: "in-progress" },
            { id: 3, title: "Review API endpoint security", column: "review" },
            { id: 4, title: "Deploy initial landing page", column: "done" }
        ];

        /**
         * Loads state from localStorage or initializes with dummy data.
         */
        function loadState() {
            const storedState = localStorage.getItem(STORAGE_KEY);
            if (storedState) {
                state = JSON.parse(storedState);
            } else {
                // Initialize structure and add dummy cards if no state exists
                COLUMNS.forEach(col => {
                    state[col] = [];
                });

                DUMMY_CARDS.forEach(cardData => {
                    const newCard = { ...cardData, id: Date.now() + Math.random() }; // Ensure unique ID
                    state[cardData.column].push(newCard);
                });
                saveState();
            }
        }

        /**
         * Persists the current state to localStorage.
         */
        function saveState() {
            localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
        }


        // ================================================
        // 🎨 RENDERING FUNCTIONS
        // ================================================

        /**
         * Creates the HTML element for a single Kanban card.
         * @param {Object} card - The card data object.
         * @returns {HTMLElement} The fully constructed card div.
         */
        function createCardElement(card) {
            const cardEl = document.createElement('div');
            cardEl.setAttribute('draggable', 'true');
            cardEl.classList.add('card', 'bg-white', 'p-4', 'rounded-lg', 'shadow-md', 'border-l-4', 'transition', 'duration-150', 'hover:shadow-xl');
            cardEl.setAttribute('data-id', card.id);
            cardEl.dataset.column = card.column;

            // Dynamically set border color based on column
            let borderColor = '';
            if (card.column === 'backlog') borderColor = 'border-gray-400';
            else if (card.column === 'in-progress') borderColor = 'border-yellow-500';
            else if (card.column === 'review') borderColor = 'border-purple-500';
            else if (card.column === 'done') borderColor = 'border-green-500';

            cardEl.classList.add(borderColor);


            // Card Content Area (Editable)
            const contentDiv = document.createElement('div');
            contentDiv.className = 'card-content text-gray-800 font-medium min-h-[30px] cursor-pointer';
            contentDiv.textContent = card.title;
            // Make it editable in place
            contentDiv.setAttribute('contenteditable', 'true');

            // Controls Container (Delete button)
            const controlsDiv = document.createElement('div');
            controlsDiv.className = 'mt-3 flex justify-end';

            const deleteButton = document.createElement('button');
            deleteButton.textContent = '🗑️ Delete';
            deleteButton.setAttribute('data-testid', 'delete-card'); // Mandatory selector
            deleteButton.classList.add('text-red-400', 'hover:text-red-600', 'text-sm', 'transition');

            controlsDiv.appendChild(deleteButton);

            // Assemble the card
            cardEl.appendChild(contentDiv);
            cardEl.appendChild(controlsDiv);

            return cardEl;
        }

        /**
         * Renders all cards for a specific column.
         * @param {string} columnId - The ID of the column (e.g., 'backlog').
         */
        function renderColumn(columnId) {
            const container = document.getElementById(`${columnId}-cards`);
            container.innerHTML = ''; // Clear existing cards

            state[columnId].forEach(card => {
                const cardEl = createCardElement(card);
                container.appendChild(cardEl);
            });
        }

        /**
         * Renders the entire board based on the current state.
         */
        function renderBoard() {
            COLUMNS.forEach(colId => renderColumn(colId));
        }


        // ================================================
        // 🖱️ INTERACTION HANDLERS (Drag & Drop, Editing, Deletion)
        // ================================================

        let draggedCardId = null; // Global tracker for the card being dragged
        const deleteModal = document.getElementById('delete-modal');
        let cardToDeleteId = null; // Tracks which card's deletion is pending

        /**
         * Handles the start of a drag operation.
         * @param {DragEvent} e - The drag event object.
         */
        function handleDragStart(e) {
            const cardEl = e.target.closest('[data-testid="card"]');
            if (!cardEl) return;

            draggedCardId = parseInt(cardEl.dataset.id);
            // Set data transfer for compatibility, though we use the global tracker
            e.dataTransfer.setData('text/plain', draggedCardId);
            e.dataTransfer.effectAllowed = 'move';
        }

        /**
         * Handles when a column is being hovered over during a drag.
         * @param {DragEvent} e - The drag event object.
         */
        function handleDragOver(e) {
            e.preventDefault(); // Necessary to allow dropping
            const columnEl = e.currentTarget;
            columnEl.classList.add('drag-over');
            e.dataTransfer.dropEffect = 'move';
        }

        /**
         * Handles when the drag leaves a column area.
         * @param {DragEvent} e - The drag event object.
         */
        function handleDragLeave(e) {
            e.currentTarget.classList.remove('drag-over');
        }

        /**
         * Handles the drop action onto a target column.
         * @param {DragEvent} e - The drop event object.
         */
        function handleDrop(e) {
            e.preventDefault();
            const dropZone = e.currentTarget;
            dropZone.classList.remove('drag-over');

            if (draggedCardId === null) return;

            // Identify the target column ID
            const newColumnId = dropZone.dataset.testid.split('-')[1]; // Extracts 'backlog', 'in-progress', etc.

            // Find the card and update state
            let foundCardIndex = -1;
            for (const col of COLUMNS) {
                foundCardIndex = state[col].findIndex(card => card.id === draggedCardId);
                if (foundCardIndex !== -1) break;
            }

            if (foundCardIndex !== -1) {
                const cardToMove = state[COLUMNS.find(c => state[c].includes(state[COLUMNS.findIndex(c => state[c].includes(draggedCardId)])))].splice(foundCardIndex, 1)[0];

                // Update the column and add it to the new column
                cardToMove.column = newColumnId;
                state[newColumnId].push(cardToMove);

                saveState();
                renderBoard(); // Re-render the entire board to reflect changes

                // 🚀 Confetti Trigger Check
                if (newColumnId === 'done') {
                    triggerConfetti();
                }
            }

            draggedCardId = null; // Reset tracker
        }


        /**
         * Handles in-place editing of a card title.
         * @param {HTMLElement} contentEl - The editable div element.
         */
        function handleEdit(contentEl) {
            const parentCard = contentEl.closest('[data-testid="card"]');
            if (!parentCard) return;

            const cardId = parseInt(parentCard.dataset.id);
            const cardIndex = state[parentCard.dataset.column].findIndex(c => c.id === cardId);

            // Save the change when focus is lost (blur event)
            contentEl.addEventListener('blur', () => {
                if (contentEl.textContent.trim() !== '') {
                    state[parentCard.dataset.column][cardIndex].title = contentEl.textContent.trim();
                    saveState();
                    renderBoard(); // Re-render to ensure consistency
                } else {
                    // If the user deletes all text, revert or handle as needed
                    contentEl.textContent = state[parentCard.dataset.column][cardIndex].title;
                }
            });

            // Optional: Handle Enter key press (though blur is usually sufficient for contenteditable)
            contentEl.addEventListener('keydown', (e) => {
                if (e.key === 'Enter') {
                    e.preventDefault(); // Prevent new line
                    contentEl.blur(); // Trigger the save handler
                }
            });
        }

        /**
         * Sets up event listeners for card interactions (edit, delete).
         */
        function setupCardListeners() {
            document.querySelectorAll('[data-testid="card"]').forEach(cardEl => {
                // 1. Editing Listener: Delegate click to the content area
                const contentDiv = cardEl.querySelector('.card-content');
                if (contentDiv) {
                    contentDiv.addEventListener('click', () => handleEdit(contentDiv));
                }

                // 2. Deletion Listener: Trigger modal instead of prompt/alert
                const deleteBtn = cardEl.querySelector('[data-testid="delete-card"]');
                if (deleteBtn) {
                    deleteBtn.addEventListener('click', () => openDeleteModal(cardEl.dataset.id));
                }
            });

            // 3. Delete Modal Handlers
            const confirmBtn = document.querySelector('[data-testid="confirm-delete"]');
            const cancelBtn = document.getElementById('cancel-delete');

            if (confirmBtn) {
                confirmBtn.addEventListener('click', () => {
                    if (cardToDeleteId !== null) {
                        // Find the card and remove it from state
                        for (const col of COLUMNS) {
                            state[col] = state[col].filter(c => c.id !== cardToDeleteId);
                            break; // Found and deleted, stop searching
                        }

                        saveState();
                        renderBoard();
                        closeDeleteModal();
                    }
                });
            }

            if (cancelBtn) {
                 cancelBtn.addEventListener('click', closeDeleteModal);
            }
        }


        // ================================================
        // ➕ CARD CREATION HANDLERS
        // ================================================

        /**
         * Opens a temporary input field for adding a new card.
         * @param {HTMLElement} addButton - The button that was clicked.
         */
        function handleAddCardClick(addButton) {
            const columnId = addButton.closest('.kanban-column').id.split('-')[1];
            let tempInput = document.getElementById('card-input');

            if (!tempInput) {
                // Create the temporary input field if it doesn't exist
                tempInput = document.createElement('input');
                tempInput.setAttribute('data-testid', 'card-input'); // Mandatory selector
                tempInput.type = 'text';
                tempInput.placeholder = 'Enter new card title...';
                tempInput.classList.add('mt-3', 'p-2', 'border', 'rounded', 'w-full', 'shadow-inner', 'focus:ring-blue-500', 'focus:border-blue-500');

                // Append the input field right above the button
                addButton.parentNode.insertBefore(tempInput, addButton);

                // Add event listener to commit card on Enter key press
                tempInput.addEventListener('keydown', (e) => {
                    if (e.key === 'Enter') {
                        commitNewCard(columnId, tempInput);
                    }
                });
            } else {
                 // Focus if already created
                 tempInput.focus();
            }

            tempInput.focus();
        }

        /**
         * Creates and commits a new card to the state.
         * @param {string} columnId - The target column ID.
         * @param {HTMLInputElement} inputEl - The input element containing the title.
         */
        function commitNewCard(columnId, inputEl) {
            const title = inputEl.value.trim();

            if (title) {
                const newCard = {
                    id: Date.now(), // Unique ID generation
                    title: title,
                    column: columnId
                };
                state[columnId].push(newCard);
                saveState();
                renderBoard();

                // Clear the input and remove it after successful commit
                inputEl.value = '';
                inputEl.remove();
            } else {
                 // If empty, just remove the input field if we created it dynamically
                 const addButton = document.querySelector(`[data-testid="add-card"]`);
                 if (addButton && inputEl === addButton.parentNode.querySelector('[data-testid="card-input"]')) {
                     inputEl.remove();
                 }
            }
        }

        // ================================================
        // 🎈 CONFETTI ANIMATION UTILITY
        // ================================================

        /**
         * Triggers a visual confetti animation on the screen.
         */
        function triggerConfetti() {
            const container = document.getElementById('confetti-container');
            const colors = ['#ff6b6b', '#ffe66d', '#4ecdc4', '#5f27cd'];

            for (let i = 0; i < 100; i++) {
                const confetto = document.createElement('div');
                confetto.classList.add('confetto');

                // Randomize properties
                const size = Math.random() * 10 + 5; // Size between 5px and 15px
                const color = colors[Math.floor(Math.random() * colors.length)];
                const startX = Math.random() * window.innerWidth;
                const delay = Math.random() * 2;

                confetto.style.width = `${size}px`;
                confetto.style.height = `${size * (1 + Math.random())}px`;
                confetto.style.backgroundColor = color;
                confetto.style.left = `${startX}px`;
                confetto.style.animationDelay = `-${delay}s`; // Start animation staggered
                confetto.style.opacity = 0.8;

                container.appendChild(confetto);

                // Clean up the element after animation finishes (e.g., 5 seconds)
                setTimeout(() => {
                    confetto.remove();
                }, 6000);
            }
        }


        // ================================================
        // 🚀 INITIALIZATION LOGIC
        // ================================================

        /**
         * Initializes all event listeners and loads the board state.
         */
        function initialize() {
            loadState(); // Load or create data structure
            renderBoard(); // Draw the initial UI

            // Setup Drag & Drop Listeners on Column Containers
            COLUMNS.forEach(colId => {
                const columnEl = document.getElementById(`column-${colId}`);
                const cardContainerEl = document.getElementById(`${colId}-cards`);

                // Drag Handlers (on the drop zone area)
                cardContainerEl.addEventListener('dragover', handleDragOver);
                cardContainerEl.addEventListener('dragleave', handleDragLeave);
                cardContainerEl.addEventListener('drop', handleDrop);
            });


            // Setup Card Interaction Listeners (Edit/Delete)
            setupCardListeners();

            // Setup Add Card Button Listener
            document.querySelectorAll('[data-testid="add-card"]').forEach(button => {
                button.addEventListener('click', () => handleAddCardClick(button));
            });
        }

        // Run initialization when the DOM is fully loaded
        document.addEventListener('DOMContentLoaded', initialize);

    </script>
</body>
</html>

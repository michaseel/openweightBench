    /* ================================================================
       NAMESPACE & STORAGE

    const STORAGE_KEY = 'kanban-board-v1';

    /**
     * Load board state from localStorage, or return null if nothing stored.
     */
    function loadState() {
      try {
        const raw = localStorage.getItem(STORAGE_KEY);
        return raw ? JSON.parse(raw) : null;
      } catch {
        return null;
      }
    }

    /**
     * Persist the current board state to localStorage.
     */
    function saveState(state) {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
    }

    /* ================================================================
       DUMMY DATA
     ================================================================ */

    /**
     * Generate a unique ID (simple timestamp + random suffix).

    function generateId() {
      return 'card-' + Date.now().toString(36) + '-' + Math.random().toString(36).slice(2, 8);
    }

    /* ================================================================
       STATE

    // Current board state — loaded from storage or defaults
    let boardState = loadState() || getDefaultState();
    saveState(boardState);

    // Track which card is being dragged

    // Track which card is pending deletion (for the modal)

    /* ================================================================

     ================================================================ */
    const boardEl = document.getElementById('board');
    const deleteModal = document.getElementById('delete-modal');
    const modalConfirm = document.getElementById('modal-confirm');
    const modalCancel = document.getElementById('modal-cancel');

    /* ================================================================
       RENDERING
     ================================================================ */

    /**
     * Render the entire board from current state.
     */
    function render() {
      boardEl.innerHTML = '';
      boardState.columns.forEach((col) => {
        boardEl.appendChild(createColumnElement(col));
      });
    }

    /**
     * Create the DOM element for a single column.
     */
    function createColumnElement(col) {
      // --- Column wrapper ---
      const colEl = document.createElement('div');
      colEl.className = 'flex-1 w-full min-w-[280px] max-w-[380px]';
      colEl.setAttribute('data-column-id', col.id);

      // --- Column container (drop zone) ---

      container.style.backgroundColor = col.bgAccent;
      container.setAttribute('data-testid', `column-${col.id}`);
      container.setAttribute('data-column-id', col.id);

      // Drag-and-drop events on the container
      container.addEventListener('dragover', handleDragOver);
      container.addEventListener('dragenter', handleDragEnter);
      cardrop', handleDragLeave);
      container.addEventListener('drop', handleDrop);

      // --- Header ---
      const header = document.createElement('div');
      header.className = 'flex items-center justify-between mb-4';

      // Title + badge
      const titleWrap = document.createElement('div');
      titleWrap.className = 'flex items-center gap-3';

      const gradientSpan = document.createElement('span');
      gradientSpan.className = `inline-block w-3 h-3 rounded-full bg-gradient-to-r ${col.color}`;

      const titleSpan = document.createElement('h2');
      titleSpan.className = 'text-lg font-bold text-white tracking-wide';
      titleSpan.textContent = col.title;

      titleWrap.appendChild(gradientSpan);
      titleWrap.appendChild(titleSpan);
      titleWrap.appendChild(badge);

      addBtn.setAttribute('data-testid', 'add-card');

      addBtn.textContent = '+';
      addBtn.title = 'Add card';
      addBtn.addEventListener('click', () => showAddForm(container, col));

      header.appendChild(titleWrap);
      header.appendChild(addBtn);
      container.appendChild(header);

      // --- Cards list ---

    /**
     * Create the DOM element for a single card.
     */
    function createCardElement(card, columnId) {
      const cardEl = document.createElement('div');
      cardEl.className = 'kanban-card card-enter bg-white/10 backdrop-blur-sm border border-white/10 rounded-xl p-4 group';
      cardEl.setAttribute('data-testid', 'card');
      cardEl.setAttribute('data-card-id', card.id);
      cardEl.setAttribute('draggable', 'true');

      // --- Card content area (text + delete button) ---
      const contentRow = document.createElement('div');
      contentRow.className = 'flex items-start justify-between gap-2';

      const textEl = document.createElement('p');
      textEl.className = 'text-sm text-slate-200 leading-relaxed flex-1 break-words';
      textEl.textContent = card.title;

      // Delete button (visible on hover)
      const deleteBtn = document.createElement('button');
      deleteBtn.setAttribute('data-testid', 'delete-card');
      deleteBtn.className = 'opacity-0 group-hover:opacity-100 transition-opacity duration-200 text-slate-500 hover:text-red-400 text-lg cursor-pointer p-0.5 rounded';
      deleteBtn.innerHTML = '✕';
      deleteBtn.title = 'Delete card';
      deleteBtn.addEventListener('click', (e) => {
        e.stopPropagation();
        openDeleteModal(card.id, columnId);
      });

      contentRow.appendChild(textEl);
      contentRow.appendChild(deleteBtn);
      cardEl.appendChild(contentRow);

      // --- Double-click to edit ---
      textEl.addEventListener('dblclick', (e) => {
        e.stopPropagation();
        startEditCard(cardEl, card, columnId, textEl);
      });

      // --- Drag events ---
      cardEl.addEventListener('dragstart', (e) => {
        draggedCardId = card.id;
        draggedSourceColumnId = columnId;
        cardEl.classList.add('dragging');
        e.dataTransfer.effectAllowed = 'move';
        // Set drag image to transparent so we use CSS transform instead
        const ghost = document.createElement('div');
        ghost.style.opacity = '0';
        document.body.appendChild(ghost);
        e.dataTransfer.setDragImage(ghost, 0, 0);
        setTimeout(() => document.body.removeChild(ghost), 0);
      });

      cardEl.addEventListener('dragend', () => {
        cardEl.classList.remove('dragging');
        draggedCardId = null;
        draggedSourceColumnId = null;
        // Remove all drag-over highlights
        document.querySelectorAll('.drag-over').forEach((el) => el.classList.remove('drag-over'));
      });

      return cardEl;
    }

    /* ================================================================
       ADD CARD
     ================================================================ */

    /**
     * Show an inline form at the bottom of a column to add a new card.
     */
    function showAddForm(containerEl, column) {
      // Remove any existing add form in this column
      const existing = containerEl.querySelector('.add-card-form');
      if (existing) existing.remove();

      const form = document.createElement('div');
      form.className = 'add-card-form mt-3';

      const input = document.createElement('input');
      input.type = 'text';
      input.setAttribute('data-testid', 'card-input');
      input.className = 'card-edit-input';
      input.placeholder = 'Card title…';
      input.maxLength = 200;

      const actions = document.createElement('div');
      actions.className = 'flex gap-2 mt-2';

      const saveBtn = document.createElement('button');
      saveBtn.className = 'px-3 py-1.5 rounded-lg bg-violet-600 hover:bg-violet-500 text-white text-xs font-semibold transition-colors duration-200';
      saveBtn.textContent = 'Add';

      const cancelBtn = document.createElement('button');
      cancelBtn.className = 'px-3 py-1.5 rounded-lg bg-slate-600 hover:bg-slate-500 text-slate-300 text-xs font-semibold transition-colors duration-200';
      cancelBtn.textContent = 'Cancel';

      actions.appendChild(saveBtn);
      actions.appendChild(cancelBtn);
      form.appendChild(input);
      form.appendChild(actions);
      containerEl.appendChild(form);

      // Focus the input
      input.focus();

      /**
       * Commit the new card.
       */
      function commit() {
        const title = input.value.trim();
        if (!title) return; // ignore empty

        const newCard = { id: generateId(), title, created: Date.now() };
        column.cards.push(newCard);
        saveState(boardState);
        render();
      }

      // Enter key commits
      input.addEventListener('keydown', (e) => {
        if (e.key === 'Enter') {
          e.preventDefault();
          commit();
        }
        if (e.key === 'Escape') {
          form.remove();
        }
      });

      saveBtn.addEventListener('click', commit);
      cancelBtn.addEventListener('click', () => form.remove());
    }

    /* ================================================================
       EDIT CARD IN PLACE
     ================================================================ */

    /**
     * Replace the card text with an editable input. Double-click triggers this.
     */
    function startEditCard(cardEl, card, columnId, textEl) {
      const input = document.createElement('input');
      input.type = 'text';
      input.className = 'card-edit-input';
      input.value = card.title;
      input.maxLength = 200;

      textEl.replaceWith(input);
      input.focus();
      input.select();

      function finishEdit() {
        const newTitle = input.value.trim();
        if (newTitle) {
          card.title = newTitle;
          saveState(boardState);
        }
        render();
      }

      input.addEventListener('keydown', (e) => {
        if (e.key === 'Enter') {
          e.preventDefault();
          finishEdit();
        }
        if (e.key === 'Escape') {
          render(); // revert
        }
      });

      input.addEventListener('blur', finishEdit);
    }

    /* ================================================================
       DELETE CARD (Modal)
     ================================================================ */

    /**
     * Open the delete-confirmation modal.
     */
    function openDeleteModal(cardId, columnId) {
      pendingDeleteCardId = cardId;
      pendingDeleteColumnId = columnId;
      deleteModal.classList.add('active');
    }

    /**
     * Close the modal without deleting.
     */
    function closeDeleteModal() {
      deleteModal.classList.remove('active');
      pendingDeleteCardId = null;
      pendingDeleteColumnId = null;
    }

    /**
     * Confirm deletion: remove the card, save, re-render.
     */
    function confirmDelete() {
      if (!pendingDeleteCardId || !pendingDeleteColumnId) return;

      const col = boardState.columns.find((c) => c.id === pendingDeleteColumnId);
      if (!col) { closeDeleteModal(); return; }

      col.cards = col.cards.filter((c) => c.id !== pendingDeleteCardId);
      saveState(boardState);
      closeDeleteModal();
      render();
    }

    modalConfirm.addEventListener('click', confirmDelete);
    modalCancel.addEventListener('click', closeDeleteModal);

    // Close modal on overlay click
    deleteModal.addEventListener('click', (e) => {
      if (e.target === deleteModal) closeDeleteModal();
    });

    // Close modal on Escape
    document.addEventListener('keydown', (e) => {
      if (e.key === 'Escape' && deleteModal.classList.contains('active')) {
        closeDeleteModal();
      }
    });

    /* ================================================================
       DRAG & DROP
     ================================================================ */

    /**
     * handleDragOver — allow drop by preventing default.
     */
    function handleDragOver(e) {
      e.preventDefault();
      e.dataTransfer.dropEffect = 'move';
    }

    /**
     * handleDragEnter — highlight the drop zone.
     */
    function handleDragEnter(e) {
      e.preventDefault();
      const col = e.currentTarget;
      col.classList.add('drag-over');
    }

    /**
     * handleDragLeave — remove highlight when leaving.
     */
    function handleDragLeave(e) {
      // Only remove if we actually left the container (not entering a child)
      if (!e.currentTarget.contains(e.relatedTarget)) {
        e.currentTarget.classList.remove('drag-over');
      }
    }

    /**
     * handleDrop — move the dragged card to the target column.
     */
    function handleDrop(e) {
      e.preventDefault();
      const targetColEl = e.currentTarget;
      targetColEl.classList.remove('drag-over');

      if (!draggedCardId || !draggedSourceColumnId) return;

      const targetColumnId = targetColEl.getAttribute('data-column-id');

      // Same column — no move needed
      if (targetColumnId === draggedSourceColumnId) return;

      // Find source & target columns
      const sourceCol = boardState.columns.find((c) => c.id === draggedSourceColumnId);
      const targetCol = boardState.columns.find((c) => c.id === targetColumnId);
      if (!sourceCol || !targetCol) return;

      // Find the card in source
      const cardIndex = sourceCol.cards.findIndex((c) => c.id === draggedCardId);
      if (cardIndex === -1) return;

      // Remove from source
      const [movedCard] = sourceCol.cards.splice(cardIndex, 1);

      // Add to target
      targetCol.cards.push(movedCard);

      // Persist
      saveState(boardState);

      // Trigger confetti if dropped into "Done"
      if (targetColumnId === 'done') {
        launchConfetti();
      }

      // Re-render
      render();
    }

    /* ================================================================
       CONFETTI ANIMATION
     ================================================================ */

    const confettiCanvas = document.getElementById('confetti-canvas');
    const confettiCtx = confettiCanvas.getContext('2d');
    let confettiParticles = [];
    let confettiAnimating = false;

    /**
     * Resize the confetti canvas to fill the viewport.
     */
    function resizeConfettiCanvas() {
      confettiCanvas.width = window.innerWidth;
      confettiCanvas.height = window.innerHeight;
    }
    window.addEventListener('resize', resizeConfettiCanvas);
    resizeConfettiCanvas();

    /**
     * Launch a burst of confetti particles.
     */
    function launchConfetti() {
      confettiParticles = [];
      const colors = [
        '#f43f5e', '#8b5cf6', '#3b82f6', '#10b981', '#f59e0b',
        '#ec4899', '#06b6d4', '#fbbf24', '#a78bfa', '#34d399',
      ];

      // Create 150 particles
      for (let i = 0; i < 150; i++) {
        confettiParticles.push({
          x: Math.random() * confettiCanvas.width,
          y: -20 - Math.random() * 200, // start above viewport
          w: Math.random() * 10 + 5,
          h: Math.random() * 6 + 3,
          color: colors[Math.floor(Math.random() * colors.length)],
          vx: (Math.random() - 0.5) * 4,
          vy: Math.random() * 3 + 2,
          rotation: Math.random() * 360,
          rotationSpeed: (Math.random() - 0.5) * 10,
          opacity: 1,
          decay: 0.003 + Math.random() * 0.005,
        });
      }

      if (!confettiAnimating) {
        confettiAnimating = true;
        animateConfetti();
      }
    }

    /**
     * Animation loop for confetti particles.
     */
    function animateConfetti() {
      confettiCtx.clearRect(0, 0, confettiCanvas.width, confettiCanvas.height);

      confettiParticles = confettiParticles.filter((p) => p.opacity > 0);

      if (confettiParticles.length === 0) {
        confettiAnimating = false;
        return;
      }

      confettiParticles.forEach((p) => {
        p.x += p.vx;
        p.y += p.vy;
        p.vy += 0.05; // gravity
        p.rotation += p.rotationSpeed;
        p.opacity -= p.decay;

        confettiCtx.save();
        confettiCtx.translate(p.x, p.y);
        confettiCtx.rotate((p.rotation * Math.PI) / 180);
        confettiCtx.globalAlpha = Math.max(0, p.opacity);
        confettiCtx.fillStyle = p.color;
        confettiCtx.fillRect(-p.w / 2, -p.h / 2, p.w, p.h);
        confettiCtx.restore();
      });

      requestAnimationFrame(animateConfetti);
    }

    /* ================================================================
       INITIAL RENDER
     ================================================================ */
    render();
  </script>
</body>

This single HTML file delivers a complete, production-quality Kanban board. Here's a summary of what's included:

### Features
| Feature | Implementation |
|---|---|
| **4 Columns** | Backlog, In Progress, Review, Done — with gradient-colored headers and card count badges |
| **Drag & Drop** | HTML5 DnD API with visual feedback (highlighted drop zones, card rotation while dragging) |
| **Inline Editing** | Double-click any card text to edit in place; press Enter to save, Escape to cancel |
| **localStorage** | State persisted under namespace `kanban-board-v1`; survives page reloads |
| **Delete with Confirmation** | Custom modal (no `alert`/`prompt`); backdrop blur, animated scale-in |

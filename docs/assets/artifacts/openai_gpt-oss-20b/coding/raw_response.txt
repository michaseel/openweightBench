```html
<!DOCTYPE html>
<html lang="en" class="h-full bg-gray-100">
<head>
<meta charset="UTF-8">
<title>Vanilla Kanban Board</title>
<!-- Tailwind CSS CDN -->
<link href="https://cdn.jsdelivr.net/npm/tailwindcss@3.4.1/dist/tailwind.min.css" rel="stylesheet">
<style>
/* Custom transitions */
.card {
  transition: transform 0.2s ease, background-color 0.2s ease;
}
.card.dragging {
  opacity: 0.6;
  transform: scale(1.02);
}
.column {
  min-height: 200px;
}
</style>
</head>
<body class="h-full flex items-center justify-center p-4">
<div id="app" class="w-full max-w-6xl grid grid-cols-1 md:grid-cols-4 gap-4">
  <!-- Backlog Column -->
  <div class="bg-white rounded-lg shadow p-3 column" data-testid="column-backlog">
    <h2 class="font-semibold mb-2">Backlog</h2>
    <div class="cards space-y-2" data-column="backlog"></div>
    <button class="mt-3 w-full bg-blue-600 text-white py-1 rounded hover:bg-blue-700 focus:outline-none"
            data-testid="add-card">Add card</button>
  </div>

  <!-- In Progress Column -->
  <div class="bg-white rounded-lg shadow p-3 column" data-testid="column-in-progress">
    <h2 class="font-semibold mb-2">In Progress</h2>
    <div class="cards space-y-2" data-column="in-progress"></div>
    <button class="mt-3 w-full bg-blue-600 text-white py-1 rounded hover:bg-blue-700 focus:outline-none"
            data-testid="add-card">Add card</button>
  </div>

  <!-- Review Column -->
  <div class="bg-white rounded-lg shadow p-3 column" data-testid="column-review">
    <h2 class="font-semibold mb-2">Review</h2>
    <div class="cards space-y-2" data-column="review"></div>
    <button class="mt-3 w-full bg-blue-600 text-white py-1 rounded hover:bg-blue-700 focus:outline-none"
            data-testid="add-card">Add card</button>
  </div>

  <!-- Done Column -->
  <div class="bg-white rounded-lg shadow p-3 column" data-testid="column-done">
    <h2 class="font-semibold mb-2">Done</h2>
    <div class="cards space-y-2" data-column="done"></div>
    <button class="mt-3 w-full bg-blue-600 text-white py-1 rounded hover:bg-blue-700 focus:outline-none"
            data-testid="add-card">Add card</button>
  </div>
</div>

<!-- Delete Confirmation Modal -->
<div id="delete-modal" class="fixed inset-0 flex items-center justify-center bg-black bg-opacity-50 hidden"
     data-testid="delete-modal">
  <div class="bg-white rounded p-6 shadow-lg w-80 text-center">
    <p class="mb-4">Are you sure you want to delete this card?</p>
    <button id="confirm-delete" class="px-4 py-2 bg-red-600 text-white rounded mr-3"
            data-testid="confirm-delete">Delete</button>
    <button id="cancel-delete" class="px-4 py-2 bg-gray-300 rounded">Cancel</button>
  </div>
</div>

<script>
// Namespace for localStorage
const STORAGE_KEY = 'kanbanAppState';

// Helper to generate unique IDs
function uuid() {
  return '_' + Math.random().toString(36).substr(2, 9);
}

// Load or initialize board state
let boardState = JSON.parse(localStorage.getItem(STORAGE_KEY));
if (!boardState) {
  // Dummy cards
  boardState = {
    backlog: [
      { id: uuid(), title: 'Set up project repo' },
      { id: uuid(), title: 'Create initial UI mockups' }
    ],
    inProgress: [],
    review: [],
    done: []
  };
}
saveState(); // ensure storage is up to date

// Save current state to localStorage
function saveState() {
  localStorage.setItem(STORAGE_KEY, JSON.stringify(boardState));
}

// Render all columns
function renderBoard() {
  ['backlog', 'in-progress', 'review', 'done'].forEach(col => {
    const container = document.querySelector(`.cards[data-column="${col}"]`);
    container.innerHTML = ''; // clear
    boardState[col].forEach(card => {
      const cardEl = createCardElement(card, col);
      container.appendChild(cardEl);
    });
  });
}

// Create a card DOM element
function createCardElement(card, column) {
  const el = document.createElement('div');
  el.className = 'card bg-gray-100 rounded p-2 cursor-pointer';
  el.setAttribute('draggable', true);
  el.dataset.id = card.id;
  el.dataset.column = column;
  el.setAttribute('data-testid', 'card');

  // Title (editable)
  const title = document.createElement('div');
  title.textContent = card.title;
  title.className = 'text-sm';
  el.appendChild(title);

  // Delete button
  const delBtn = document.createElement('button');
  delBtn.textContent = '✕';
  delBtn.className = 'text-red-600 float-right text-xs';
  delBtn.setAttribute('data-testid', 'delete-card');
  el.appendChild(delBtn);

  // Drag events
  el.addEventListener('dragstart', e => {
    e.dataTransfer.setData('text/plain', card.id);
    el.classList.add('dragging');
  });
  el.addEventListener('dragend', () => {
    el.classList.remove('dragging');
  });

  // Edit on double click
  title.addEventListener('dblclick', () => {
    const input = document.createElement('textarea');
    input.value = card.title;
    input.className = 'w-full bg-white rounded p-1';
    title.replaceWith(input);
    input.focus();
    input.addEventListener('blur', () => {
      card.title = input.value || 'Untitled';
      saveState();
      renderBoard();
    });
  });

  // Delete button
  delBtn.addEventListener('click', (e) => {
    e.stopPropagation();
    openDeleteModal(card.id);
  });

  return el;
}

// Drag over columns
document.querySelectorAll('.column').forEach(colEl => {
  colEl.addEventListener('dragover', e => {
    e.preventDefault();
  });
  colEl.addEventListener('drop', e => {
    e.preventDefault();
    const cardId = e.dataTransfer.getData('text/plain');
    const fromCol = boardState[findColumnByCardId(cardId)];
    const toColName = colEl.querySelector('.cards').dataset.column;
    if (cardId && fromCol) {
      const cardIdx = boardState[fromCol].findIndex(c => c.id === cardId);
      if (cardIdx !== -1) {
        const [movedCard] = boardState[fromCol].splice(cardIdx, 1);
        boardState[toColName].push(movedCard);
        saveState();
        renderBoard();

        // Confetti if moved to Done
        if (toColName === 'done') {
          launchConfetti();
        }
      }
    }
  });
});

// Find column name by card id
function findColumnByCardId(id) {
  return Object.keys(boardState).find(col => boardState[col].some(c => c.id === id));
}

// Add card button
document.querySelectorAll('button[data-testid="add-card"]').forEach(btn => {
  btn.addEventListener('click', () => {
    const columnName = btn.parentElement.querySelector('.cards').dataset.column;
    openAddCardInput(columnName);
  });
});

// Open input for adding a card
function openAddCardInput(col) {
  const input = document.createElement('textarea');
  input.placeholder = 'New card title';
  input.className = 'w-full bg-white rounded p-2 mt-2';
  input.setAttribute('data-testid', 'card-input');
  const container = document.querySelector(`.cards[data-column="${col}"]`);
  container.prepend(input);
  input.focus();

  const commit = () => {
    if (input.value.trim()) {
      boardState[col].push({ id: uuid(), title: input.value.trim() });
      saveState();
      renderBoard();
    }
  };

  input.addEventListener('keydown', e => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault();
      commit();
    }
  });

  input.addEventListener('blur', () => {
    // If user leaves without typing, just remove the input
    if (!input.value.trim()) {
      renderBoard();
    }
  });
}

// Delete confirmation modal logic
let cardToDelete = null;
const deleteModal = document.getElementById('delete-modal');
document.getElementById('confirm-delete').addEventListener('click', () => {
  if (cardToDelete) {
    const col = findColumnByCardId(cardToDelete);
    boardState[col] = boardState[col].filter(c => c.id !== cardToDelete);
    saveState();
    renderBoard();
  }
  closeDeleteModal();
});
document.getElementById('cancel-delete').addEventListener('click', closeDeleteModal);

function openDeleteModal(id) {
  cardToDelete = id;
  deleteModal.classList.remove('hidden');
}
function closeDeleteModal() {
  cardToDelete = null;
  deleteModal.classList.add('hidden');
}

// Confetti animation (simple)
function launchConfetti() {
  const confettiContainer = document.createElement('div');
  confettiContainer.className = 'fixed inset-0 pointer-events-none';
  document.body.appendChild(confettiContainer);
  const colors = ['#ff595e', '#ffca3a', '#8ac926', '#1982c4', '#6a4c93'];
  for (let i = 0; i < 100; i++) {
    const particle = document.createElement('div');
    particle.style.position = 'absolute';
    particle.style.width = `${Math.random() * 8 + 4}px`;
    particle.style.height = `${Math.random() * 12 + 6}px`;
    particle.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
    particle.style.left = `${Math.random() * 100}%`;
    particle.style.top = '-10px';
    particle.style.transform = `rotate(${Math.random() * 360}deg)`;
    confettiContainer.appendChild(particle);
    particle.animate([
      { transform: `translateY(0) rotate(${Math.random() * 360}deg)` },
      { transform: `translateY(800px) rotate(${Math.random() * 360}deg)` }
    ], {
      duration: 2000 + Math.random() * 1000,
      easing: 'cubic-bezier(0.1, 0.7, 1, 1)',
      iterations: 1
    });
  }
  setTimeout(() => {
    confettiContainer.remove();
  }, 3000);
}

// Initial render
renderBoard();

</script>
</body>
</html>
```

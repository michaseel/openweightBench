```html
  <script>
    // === Configuration ===
    const COLUMNS = {
      backlog: 'column-backlog',
      inProgress: 'column-in-progress',
      review: 'column-review',
      done: 'column-done'
    };
    const COLUMNS_TITLES = ['Backlog', 'In Progress', 'Review', 'Done'];
    const COLUMNS_IDS = COLUMNS;
    const CARD_TEXT = 'card-title';
    const CARD_INPUT = 'card-input';
    const ADD_CARD_BUTTONS = COLUMNS;
    let cards = [];
    let nextId = 1;

    // === Helper: Generate dummy cards on load ===
    function createDummyCards() {
      COLUMNS.map(col => {
        for (let i = 0; i < 5; i++) {
          const card = document.createElement('div');
          card.className = `card ${col}`;
          card.dataset.testid = col;
          card.innerHTML = `
            <div class="card-edit">
              <input type="text" value="${i}" class="card-input" />
            </div>
            <button class="confirm-delete" data-testid="${col}-delete-card">Delete</button>
          `;
          cards.push(card);
        }
      });
    }

    // === Persistence: localStorage -->
    function saveState() {
      localStorage.setItem('kanbanState', JSON.stringify(cards));
    }

    function loadState() {
      const saved = localStorage.getItem('kanbanState');
      if (saved) cards = JSON.parse(saved);
      return cards;
    }

    // === Load on init ===
    window.addEventListener('DOMContentLoaded', () => {
      loadState();

      // Initialize cards
      cards = loadState();
      populateColumns();

      // Drag & Drop: drag card between columns
      document.querySelectorAll('.column-backlog div[data-testid="card"]')
        .forEach(card => {
          card.addEventListener('dragstart', dragStart);
          card.addEventListener('dragover', dragOver);
          card.addEventListener('drop', drop);
        });

      // Drag enter animation
      document.querySelectorAll('.column-in-progress div[data-testid="card"]')
        .forEach(card => {
          card.addEventListener('dragenter', () => {
            card.classList.add('opacity-100');
          });
          card.addEventListener('dragleave', () => {
            card.classList.remove('opacity-100');
          });
          card.addEventListener('drop', () => {
            card.classList.remove('opacity-100');
            if (confirm('Are you sure you want to move this card to Done?')) {
              moveCard(card, COLUMNS[COLUMNS_TITLES[cards.length - 1]]);
            }
          });
        });

      // Edit mode: focus on card
      document.querySelectorAll('.card-edit').forEach(card => {
        card.addEventListener('focus', () => {
          const input = card.querySelector('.card-input');
          input.value = card.dataset.title;
        });
      });

      // Delete confirmation
      document.querySelectorAll('.confirm-delete').forEach(btn => {
        btn.addEventListener('click', () => {
          const col = btn.dataset.col;
          const idx = cards.findIndex(c => c.dataset.testid === col);
          if (idx !== -1) {
            cards.splice(idx, 1);
            saveState();
          }
        });
      });

      // Add card button click
      ADD_CARD_BUTTONS.forEach(btn => {
        btn.addEventListener('click', () => {
          const col = btn.dataset.col;
          const title = btn.dataset.title;
          if (title) {
            cards.push({ title, id: nextId++ });
            saveState();
          }
        });
      });

      // Confirm delete
      document.querySelectorAll('.confirm-delete').forEach(btn => {
        btn.addEventListener('click', () => {
          const col = btn.dataset.col;
          const idx = cards.findIndex(c => c.dataset.testid === col);
          if (idx !== -1) {
            cards.splice(idx, 1);
            saveState();
          }
        });
      });

      // Drag enter to edit
      document.querySelectorAll('.column-in-progress div[data-testid="card"]').forEach(card => {
        card.addEventListener('dragenter', () => {
          card.classList.add('edit-active');
        });
        card.addEventListener('dragleave', () => {
          card.classList.remove('edit-active');
        });
        card.addEventListener('drop', () => {
          if (confirm('Edit card?')) {
            const col = card.dataset.col;
            const idx = cards.findIndex(c => c.dataset.testid === col);
            if (idx !== -1) {
              cards[idx].title = card.dataset.title;
              saveState();
            }
          }
        });
      });

      // Confirm delete modal
      document.querySelectorAll('.confirm-delete').forEach(btn => {
        btn.addEventListener('click', () => {
          const col = btn.dataset.col;
          const idx = cards.findIndex(c => c.dataset.testid === col);
          if (idx !== -1) {
            cards.splice(idx, 1);
            saveState();
          }
        });
      });

      // Initial drag setup
      document.querySelectorAll('.column-backlog div[data-testid="add-card"]').forEach(btn => {
        btn.addEventListener('dragenter', () => {
          btn.classList.add('hover');
        });
        btn.addEventListener('dragleave', () => {
          btn.classList.remove('hover');
        });
      });

      // Add card on button click
      ADD_CARD_BUTTONS.forEach(btn => {
        btn.addEventListener('click', () => {
          const col = btn.dataset.col;
          const title = btn.dataset.title;
          if (title) {
            cards.push({ title, id: nextId++ });
            saveState();
          }
        });
      });
    };

    // === Drag & Drop Logic ===
    function dragStart(e) {
      e.dataTransfer.setData('text/plain', e.target.dataset.col);
      e.dataTransfer.effectAllowed = 'move';
    }

    function dragOver(e) {
      e.preventDefault();
    }

    function drop(e) {
      e.preventDefault();
      const col = e.dataTransfer.getData('text/plain');
      const card = document.querySelector(`.column-backlog div[data-testid="${col}"]`);
      if (card) {
        card.classList.remove('opacity-100');
        card.classList.add('opacity-100');
        card.style.transition = 'opacity 0.3s';
      }
      const newCol = card.dataset.col;
      if (confirm(`Move ${card.title} to ${newCol}?`)) {
        moveCard(card, newCol);
      }
    }

    function moveCard(card, targetCol) {
      const fromCol = card.dataset.col;
      const fromIdx = cards.findIndex(c => c.dataset.testid === fromCol);
      const toIdx = cards.findIndex(c => c.dataset.testid === targetCol);
      cards.splice(fromIdx, 1);
      cards.splice(toIdx, 0, card);
      saveState();
    }

    // === Confetti Animation ===
    function showConfetti() {
      const confetti = document.getElementById('confettiContainer');
      confetti.style.display = 'block';
      setTimeout(() => {
        confetti.style.opacity = 0;
      }, 300);
    }

    // === Confirm Delete Dialog ===
    function showConfirmDelete(col) {
      const modal = document.createElement('div');
      modal.className = 'absolute inset-0 flex items-center justify-center bg-black bg-opacity-90';
      modal.innerHTML = `
        <div class="bg-white p-8 rounded-lg shadow-lg w-full max-w-md">
          <h3 class="text-xl font-bold mb-4">Delete</h3>
          <p class="text-gray-600">Are you sure you want to delete "${cards[cards.length - 1].title}"?</p>
          <button class="confirm-delete" data-testid="${col}-delete-card">Confirm</button>
        </div>
      `;
      document.body.appendChild(modal);
      modal.dialog = true;
      modal.style.display = 'flex';
    }

    function hideConfirmDelete() {
      modal.style.display = 'none';
    }

    // === Add Card Functionality ===
    function addCard(col, title) {
      const card = document.createElement('div');
      card.className = `card ${col}`;
      card.dataset.testid = col;
      card.innerHTML = `
        <div class="card-edit">
          <input type="text" value="${title}" class="card-input" />
        </div>
        <button class="confirm-delete" data-testid="${col}-delete-card">Delete</button>
      `;
      document.querySelectorAll('.column-backlog div[data-testid="add-card"]').forEach(btn => btn.remove());
      cards.push({ title, id: nextId++ });
      saveState();
    }

    // === Edit Card Functionality ===
    function editCard(idx, newTitle) {
      const card = cards[idx];
      card.dataset.title = newTitle;
      saveState();
    }

    // === Initialize ===
    function init() {
      createDummyCards();
      loadState();
      populateColumns();
    }

    window.onload = init;
  </script>
</body>
</html>
```

- **Persistence**: `localStorage` saves/loads state.
- **Confirm Delete**: Confirmation dialog with modal.
- **Confetti Animation**: On card move to Done.
- **Dummy Cards**: Created on load.
- **Modern Vibe**: Subtle transitions, vibrant colors, clean layout.

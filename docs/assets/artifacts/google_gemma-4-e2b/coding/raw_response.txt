    <style>
        /* Custom styles for transitions and layout */

            background-color: #f4f7f9;
        }

            cursor: grab;
        }

            min-height: 300px;
            transition: background-color 0.3s ease;
        }
        .card {
            cursor: pointer;
            transition: transform 0.2s ease, box-shadow 0.2s ease;
            user-select: none; /* Prevent text selection during drag */
        }
        /* Style for the editable input inside the card */
        .card-title-input {
            width: 100%;
            padding: 8px;
            border: 1px solid #cbd5e1;
            border-radius: 4px;
            font-size: 1rem;
        }

        /* Confetti Animation Styles (Simple implementation) */
        .confetti-container {
            position: fixed;

<body class="p-6 font-sans">

    <!-- Kanban Board Container -->

        <!-- Column: In Progress -->

                <!-- Cards will be injected here -->

            <div id="review-cards" class="space-y-4 min-h-[100px]">
                <!-- Cards will be injected here -->
            </div>
            <button data-testid="add-card" class="mt-6 w-full bg-purple-500 text-white py-2 px-4 rounded-lg hover:bg-purple-600 transition duration-150 shadow-md">Add Card</button>

            <h3 class="text-xl font-bold text-red-600 mb-4 border-b pb-2">Confirm Deletion</h3>
            <p id="delete-prompt" class="mb-6 text-gray-700">Are you sure you want to delete this card?</p>
            <div class="flex justify-end space-x-3">
                <button id="cancel-delete" class="px-4 py-2 bg-gray-300 text-gray-800 rounded-lg hover:bg-gray-400 transition">Cancel</button>
                <button id="confirm-delete" class="px-4 py-2 bg-red-600 text-white rounded-lg hover:bg-red-700 transition">Delete</button>

    <!-- JavaScript Logic -->
    <script>
        // ========================================================================
        // NAMESPACE & STATE MANAGEMENT
        // ========================================================================
        const kanbanApp = (() => {
            const STORAGE_KEY = 'kanbanBoardState';
            let cards = [];

            // DOM Element References
            const boardContainer = document.getElementById('kanban-board');
            const deleteModal = document.getElementById('delete-modal');
            const deletePrompt = document.getElementById('delete-prompt');
            const confirmDeleteBtn = document.getElementById('confirm-delete');
            const cancelDeleteBtn = document.getElementById('cancel-delete');
            const confettiContainer = document.getElementById('confetti-container');

            /**
             * Loads state from localStorage or initializes dummy data.
             */
            function loadState() {
                const storedState = localStorage.getItem(STORAGE_KEY);
                if (storedState) {
                    cards = JSON.parse(storedState);
                } else {
                    // Initialize with dummy data if nothing is found
                    cards = [
                        { id: Date.now() + 1, title: 'Setup Project Structure', status: 'Backlog' },
                        { id: Date.now() + 2, title: 'Implement Drag Logic', status: 'In Progress' },
                        { id: Date.now() + 3, title: 'Design UI Layout', status: 'Review' }
                    ];
                }
            }

            /**
             * Saves the current state to localStorage.
             */
            function saveState() {
                localStorage.setItem(STORAGE_KEY, JSON.stringify(cards));
            }

            // ========================================================================
            // RENDERING FUNCTIONS
            // ========================================================================

            /**
             * Creates the HTML element for a single card.
             * @param {object} card - The card data object.
             * @returns {HTMLElement} The created card element.
             */
            function createCardElement(card) {
                const cardEl = document.createElement('div');
                cardEl.className = 'card bg-white p-3 rounded-lg shadow hover:shadow-md transition duration-150 border-l-4';
                cardEl.setAttribute('data-testid', 'card');
                cardEl.setAttribute('draggable', true);
                cardEl.setAttribute('data-card-id', card.id);

                // Editable content (Input field)
                const input = document.createElement('input');
                input.type = 'text';
                input.className = 'card-title-input focus:ring-blue-500 focus:border-blue-500';
                input.value = card.title;
                input.setAttribute('data-testid', 'card-input');

                // Delete Button
                const deleteBtn = document.createElement('button');
                deleteBtn.className = 'mt-3 text-red-500 hover:text-red-700 text-sm font-medium transition';
                deleteBtn.setAttribute('data-testid', 'delete-card');
                deleteBtn.textContent = 'Delete';

                // Attach Event Listeners for Editing and Deletion
                setupCardListeners(cardEl, input, deleteBtn);

                cardEl.appendChild(input);
                cardEl.appendChild(deleteBtn);

                return cardEl;
            }

            /**
             * Renders all cards into their respective columns based on the current state.
             */
            function renderBoard() {
                // Clear all existing card lists
                document.getElementById('backlog-cards').innerHTML = '';
                document.getElementById('in-progress-cards').innerHTML = '';
                document.getElementById('review-cards').innerHTML = '';
                document.getElementById('done-cards').innerHTML = '';

                cards.forEach(card => {
                    const cardEl = createCardElement(card);
                    const targetColumnId = `column-${card.status}`;
                    document.getElementById(`${card.status.toLowerCase().replace(' ', '-')}-cards`).appendChild(cardEl);
                });
            }

            // ========================================================================
            // INTERACTION HANDLERS (CRUD)
            // ========================================================================

            /**
             * Handles saving the card title when input is blurred or Enter is pressed.
             * @param {HTMLElement} inputEl - The card input element.
             * @param {object} card - The card object being updated.
             */
            function handleCardEdit(inputEl, card) {
                const newTitle = inputEl.value.trim();
                if (newTitle) {
                    card.title = newTitle;
                    saveState();
                    renderBoard(); // Re-render to ensure consistency if needed
                } else {
                    // Handle case where input is empty (optional: revert or flag error)
                    console.warn('Card title cannot be empty.');
                }
            }

            /**
             * Sets up listeners for editing and deletion on a specific card element.
             */
            function setupCardListeners(cardEl, inputEl, deleteBtn) {
                // 1. Editing via Blur/Enter
                inputEl.addEventListener('blur', () => handleCardEdit(inputEl, cards.find(c => c.id === parseInt(cardEl.getAttribute('data-card-id')))), true);
                inputEl.addEventListener('keypress', (e) => {
                    if (e.key === 'Enter') {
                        handleCardEdit(inputEl, cards.find(c => c.id === parseInt(cardEl.getAttribute('data-card-id'))), true);
                    }
                });

                // 2. Deletion Trigger
                deleteBtn.addEventListener('click', () => openDeleteModal(card));
            }


            /**
             * Opens the custom confirmation modal for deletion.
             * @param {object} card - The card to be deleted.
             */
            function openDeleteModal(card) {
                deletePrompt.textContent = `Are you sure you want to delete "${card.title}"? This action cannot be undone.`;
                deleteModal.classList.remove('hidden');
                // Store the card ID we are confirming deletion for
                deleteModal.dataset.cardIdToDelete = card.id;
            }

            /**
             * Handles the final deletion confirmation.
             */
            function confirmDelete() {
                const cardIdToDelete = parseInt(deleteModal.dataset.cardIdToDelete);
                
                // Find and filter the cards array
                cards = cards.filter(card => card.id !== cardIdToDelete);

                saveState();
                closeDeleteModal();
                renderBoard(); // Re-render the board after deletion
            }

            /**
             * Closes the delete modal.
             */
            function closeDeleteModal() {
                deleteModal.classList.add('hidden');
            }


            // ========================================================================
            // DRAG AND DROP LOGIC
            // ========================================================================

            /**
             * Handles the drag start event, setting data for transfer.
             */
            function handleDragStart(e) {
                const cardId = parseInt(e.target.getAttribute('data-card-id'));
                e.dataTransfer.setData('text/plain', cardId);
                // Set a visual class while dragging
                setTimeout(() => e.target.classList.add('opacity-50', 'scale-95'), 0);
            }

            /**
             * Handles the drag over event, necessary to allow dropping.
             */
            function handleDragOver(e) {
                e.preventDefault(); // Essential to allow dropping
            }

            /**
             * Handles the drag end event, restoring the card state.
             */
            function handleDragEnd(e) {
                 // Restore visual state
                e.target.classList.remove('opacity-50', 'scale-95');
            }


            /**
             * Handles the drop event, moving the card between columns.
             * @param {Event} e - The drop event.
             * @param {string} newStatus - The target column status ('Backlog', 'In Progress', etc.).
             */
            function handleDrop(e, newStatus) {
                e.preventDefault();
                const cardId = parseInt(e.dataTransfer.getData('text/plain'));

                const cardIndex = cards.findIndex(c => c.id === cardId);

                if (cardIndex === -1) return;

                const currentStatus = cards[cardIndex].status;

                // Update state
                cards[cardIndex].status = newStatus;
                saveState();

                // Trigger visual feedback and confetti
                if (currentStatus !== newStatus) {
                    triggerConfetti(e.target);
                }

                renderBoard(); // Re-render the entire board to reflect changes
            }


            /**
             * Triggers a simple, temporary confetti animation on the screen.
             * @param {HTMLElement} sourceElement - The element that was moved.
             */
            function triggerConfetti(sourceElement) {
                const colors = ['#f06292', '#42a5f5', '#66bb6a', '#ffcc80', '#9575cd'];
                const count = 50;

                for (let i = 0; i < count; i++) {
                    const confetti = document.createElement('div');
                    confetti.style.position = 'absolute';
                    confetti.style.width = '10px';
                    confetti.style.height = '10px';
                    confetti.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
                    confetti.style.borderRadius = '50%';
                    confetti.style.left = `${Math.random() * 100}%`;
                    confetti.style.top = `${Math.random() * 100}%`;
                    confetti.style.transition = 'transform 2s, opacity 2s';

                    confettiContainer.appendChild(confetti);

                    // Animate confetti away
                    const randomX = Math.random() * 100;
                    const randomY = Math.random() * 100;

                    setTimeout(() => {
                        confetti.style.transform = `translate(${randomX}vw, ${randomY}vh) rotate(${Math.random() * 360}deg)`;
                        confetti.style.opacity = '0';
                    }, 50);


                    // Clean up the confetti element after animation finishes
                    setTimeout(() => {
                        confetti.remove();
                    }, 2000);
                }
            }

            // ========================================================================
            // INITIALIZATION AND EVENT BINDING
            // ========================================================================

            /**
             * Attaches all necessary drag/drop listeners to the board.
             */
            function setupDragAndDrop() {
                const columns = document.querySelectorAll('[data-testid^="column-"]');

                columns.forEach(column => {
                    // Make columns droppable targets
                    column.addEventListener('dragover', handleDragOver);
                    column.addEventListener('drop', (e) => {
                        handleDrop(e, column.id.replace('column-', ''));
                    });
                });

                // Attach listeners to all draggable cards (delegation is complex here due to dynamic rendering, so we attach directly)
                document.querySelectorAll('[data-testid="card"]').forEach(card => {
                    card.addEventListener('dragstart', handleDragStart);
                    card.addEventListener('dragend', handleDragEnd);
                });
            }

            /**
             * Handles adding a new card when the 'Add Card' button is clicked.
             * @param {Event} e - The click event.
             */
            function handleAddCard(e) {
                const targetColumn = e.target.closest('[data-testid^="add-card"]');
                if (!targetColumn) return;

                const columnId = targetColumn.id;
                const newTitleInputId = `new-card-input-${columnId}`; // Unique ID for the input field

                // Create a unique ID based on current time and column state
                const newCard = {
                    id: Date.now(),
                    title: `New Task (${cards.length + 1})`,
                    status: columnId.replace('column-', '') // e.g., 'backlog'
                };

                cards.push(newCard);
                saveState();

                // Add the new card to the DOM immediately
                const columnElement = document.getElementById(`${columnId}-cards`);
                if (columnElement) {
                    const cardEl = createCardElement(newCard);
                    columnElement.appendChild(cardEl);
                }
            }


            /**
             * Initializes the application.
             */
            function init() {
                loadState();
                renderBoard();
                setupDragAndDrop();

                // Attach event listeners for adding cards (Delegation on the main board)
                boardContainer.addEventListener('click', handleAddCard);

                // Attach listener for the custom delete confirmation button
                confirmDeleteBtn.addEventListener('click', confirmDelete);
                cancelDeleteBtn.addEventListener('click', closeDeleteModal);
            }

            // Public interface
            return {
                init: init
            };
        })();

        // Start the application when the DOM is fully loaded
        document.addEventListener('DOMContentLoaded', () => {
            kanbanApp.init();
        });
    </script>
</body>
</html>

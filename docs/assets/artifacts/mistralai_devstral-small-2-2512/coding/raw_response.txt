    <script>
      // Namespace for our application state in localStorage
      const KANBAN_STORAGE_KEY = 'kanban-board-data';

      // DOM Elements
      const board = document.getElementById('kanban-board');
      const columns = {
        backlog: document.querySelector('[data-testid="column-backlog"]'),
        inProgress: document.querySelector('[data-testid="column-in-progress"]'),
        review: document.querySelector('[data-testid="column-review"]'),
        done: document.querySelector('[data-testid="column-done"]')
      };

      const columnIds = ['backlog', 'inProgress', 'review', 'done'];

      // Initialize the application
      document.addEventListener('DOMContentLoaded', () => {
        loadData();
        setupEventListeners();

        // If no cards exist, create some dummy data
        if (Object.values(boardData).every(column => column.length === 0)) {
          createDummyCards();
        }
      });

      // Application State
      let boardData = {
        backlog: [],
        inProgress: [],
        review: [],
        done: []
      };

      let draggedCard = null;
      let cardToDelete = null;

      // Load data from localStorage
      function loadData() {
        try {
          const savedData = localStorage.getItem(KANBAN_STORAGE_KEY);
          if (savedData) {
            boardData = JSON.parse(savedData);
          }
        } catch (error) {
          console.error('Error loading data:', error);
          // Fallback to empty state if corrupted
          boardData = {
            backlog: [],
            inProgress: [],
            review: [],
            done: []
          };
        }
        renderBoard();
      }

      // Save data to localStorage
      function saveData() {
        try {
          localStorage.setItem(KANBAN_STORAGE_KEY, JSON.stringify(boardData));
        } catch (error) {
          console.error('Error saving data:', error);
        }
      }

      // Create dummy cards if none exist
      function createDummyCards() {
        const dummyCards = [
          { id: generateId(), title: 'Implement user authentication', column: 'backlog' },
          { id: generateId(), title: 'Design landing page', column: 'backlog' },
          { id: generateId(), title: 'Fix login button animation', column: 'inProgress' },
          { id: generateId(), title: 'Review API documentation', column: 'review' },
          { id: generateId(), title: 'Deploy to staging', column: 'done' }
        ];

        dummyCards.forEach(card => {
          boardData[card.column].push({
            id: card.id,
            title: card.title
          });
        });

        saveData();
        renderBoard();
      }

      // Generate a unique ID for cards
      function generateId() {
        return Math.random().toString(36).substring(2, 15) + Math.random().toString(36).substring(2, 15);
      }

      // Render the entire board
      function renderBoard() {
        columnIds.forEach(columnId => {
          const column = columns[columnId];
          if (!column) return;

          const contentArea = column.querySelector('.column-content');
          contentArea.innerHTML = '';

          boardData[columnId].forEach(card => {
            const cardElement = createCardElement(card, columnId);
            contentArea.appendChild(cardElement);
          });
        });
      }

      // Create a card DOM element
      function createCardElement(card, columnId) {
        const cardElement = document.createElement('div');
        cardElement.className = 'card bg-white rounded-lg shadow-sm p-4 mb-3 hover:shadow-md transition-shadow';
        cardElement.dataset.testid = 'card';
        cardElement.dataset.column = columnId;
        cardElement.dataset.cardId = card.id;
        cardElement.draggable = true;

        // Card content with edit functionality
        const cardContent = document.createElement('div');
        cardContent.className = 'flex items-start gap-3';

        const cardTitleContainer = document.createElement('div');
        cardTitleContainer.className = 'flex-1';

        const cardTitle = document.createElement('h3');
        cardTitle.className = 'font-medium text-gray-800 mb-1';
        cardTitle.textContent = card.title;

        const editIcon = document.createElement('span');
        editIcon.className = 'inline-flex items-center px-2 py-1 text-xs bg-gray-100 rounded cursor-pointer hover:bg-gray-200';
        editIcon.innerHTML = '<svg xmlns="http://www.w3.org/2000/svg" class="h-4 w-4 text-gray-600" viewBox="0 0 20 20" fill="currentColor"><path d="M13.586 3.586a2 2 0 112.828 2.828l-.793.793-2.828-2.828zM11.379 5.74a2 2 0 112.828 2.828l-.793.793-2.828-2.828zM16.05 9.42l-2.828 2.828a2 2 0 11-2.828-2.828l2.828-2.828zM19.5 10.47a2 2 0 100 3.86c-1.44 0-2.79-1.35-2.79-3s1.35-3 2.79-3c1.44 0 2.85 1.46 2.79 3zM12.05 18a2 2 0 11-4 0 2 2 0 014 0zM8.53 9.47l2.828-2.828a2 2 0 10-2.828-2.828l-2.828 2.828a2 2 0 102.828 2.828l2.828-2.828zM4 10.47a2 2 0 100-3.86c1.44 0 2.79 1.35 2.79 3s-1.35 3-2.79 3c-1.44 0-2.85-1.46-2.79-3zM6.05 18a2 2 0 114 0 2 2 0 01-4 0z"/></svg>';
        editIcon.addEventListener('click', () => startEditing(cardElement, cardTitle));

        // Delete button
        const deleteBtn = document.createElement('button');
        deleteBtn.dataset.testid = 'delete-card';
        deleteBtn.className = 'text-gray-400 hover:text-red-500 p-1 rounded-md transition-colors';
        deleteBtn.innerHTML = '<svg xmlns="http://www.w3.org/2000/svg" class="h-5 w-5" viewBox="0 0 20 20" fill="currentColor"><path fill-rule="evenodd" d="M9 2a1 1 0 00-.894.553L7.382 4H4a1 1 0 000 2v10a2 2 0 002 2h8a2 2 0 002-2V6a1 1 0 100-2h-3.382l-.724-1.447A1 1 0 0011 2H9zM7 8a1 1 0 012 0v6a1 1 0 11-2 0V8zm5-1a1 1 0 00-1 1v6a1 1 0 102 0V8a1 1 0 00-1-1z" clip-rule="evenodd"/></svg>';
        deleteBtn.addEventListener('click', (e) => {
          e.stopPropagation();
          cardToDelete = { element: cardElement, id: card.id };
          document.getElementById('delete-modal').classList.remove('hidden');
        });

        cardTitleContainer.appendChild(cardTitle);
        cardContent.appendChild(cardTitleContainer);

        // Add edit icon and delete button to the top right
        const cardActions = document.createElement('div');
        cardActions.className = 'flex flex-col gap-1';
        cardActions.appendChild(editIcon);
        cardActions.appendChild(deleteBtn);

        cardContent.appendChild(cardActions);
        cardElement.appendChild(cardContent);

        // Add drag and drop events
        cardElement.addEventListener('dragstart', (e) => {
          draggedCard = {
            element: cardElement,
            id: card.id,
            fromColumn: columnId
          };
          e.dataTransfer.effectAllowed = 'move';
          cardElement.classList.add('opacity-50');
        });

        cardElement.addEventListener('dragend', () => {
          cardElement.classList.remove('opacity-50');
        });

        // Double click to edit
        cardElement.addEventListener('dblclick', () => {
          startEditing(cardElement, cardTitle);
        });

        // Click to select (for potential future features)
        cardElement.addEventListener('click', (e) => {
          if (e.target === cardElement || e.target === cardTitleContainer || e.target === cardTitle) {
            // Highlight effect
            cardElement.classList.add('ring-2', 'ring-blue-500');
            setTimeout(() => {
              cardElement.classList.remove('ring-2', 'ring-blue-500');
            }, 300);
          }
        });

        return cardElement;
      }

      // Start editing a card's title
      function startEditing(cardElement, originalTitle) {
        const columnId = cardElement.dataset.column;
        const cardId = cardElement.dataset.cardId;

        // Find the card in our data
        const cardIndex = boardData[columnId].findIndex(card => card.id === cardId);
        if (cardIndex === -1) return;

        // Create input element
        const input = document.createElement('textarea');
        input.dataset.testid = 'card-input';
        input.className = 'w-full p-2 border border-gray-300 rounded-md resize-none focus:ring-blue-500 focus:border-blue-500';
        input.value = originalTitle.textContent;
        input.rows = 2;
        input.style.maxHeight = '100px';
        input.style.minHeight = '60px';

        // Replace title with input
        const cardContent = cardElement.querySelector('.flex.items-start.gap-3');
        if (cardContent) {
          cardContent.innerHTML = '';
          cardContent.appendChild(input);
        }

        // Focus and select text
        input.focus();
        input.select();

        // Handle Enter key to save
        const handleKeyDown = (e) => {
          if (e.key === 'Enter' && !e.shiftKey) {
            e.preventDefault();
            saveEditedCard(input, cardElement, columnId, cardIndex);
          } else if (e.key === 'Escape') {
            e.preventDefault();
            cancelEditing(cardElement, originalTitle.textContent);
          }
        };

        // Handle blur to save
        const handleBlur = () => {
          saveEditedCard(input, cardElement, columnId, cardIndex);
        };

        input.addEventListener('keydown', handleKeyDown);
        input.addEventListener('blur', handleBlur);

        // Function to save edited card
        function saveEditedCard(input, cardElement, columnId, cardIndex) {
          const newTitle = input.value.trim();
          if (newTitle === '') {
            cancelEditing(cardElement, originalTitle.textContent);
            return;
          }

          // Update the card
          boardData[columnId][cardIndex].title = newTitle;
          saveData();

          // Re-render the card
          const contentArea = columns[columnId].querySelector('.column-content');
          if (contentArea) {
            const newCard = createCardElement(boardData[columnId][cardIndex], columnId);
            contentArea.replaceChild(newCard, cardElement);
          }

          // Remove event listeners
          input.removeEventListener('keydown', handleKeyDown);
          input.removeEventListener('blur', handleBlur);
        }

        // Function to cancel editing
        function cancelEditing(cardElement, originalText) {
          const cardContent = cardElement.querySelector('.flex.items-start.gap-3');
          if (cardContent) {
            cardContent.innerHTML = `
              <div class="flex-1">
                <h3 class="font-medium text-gray-800 mb-1">${originalText}</h3>
              </div>
            `;
          }

          // Remove event listeners
          input.removeEventListener('keydown', handleKeyDown);
          input.removeEventListener('blur', handleBlur);
        }
      }

      // Set up all event listeners
      function setupEventListeners() {
        // Add card buttons
        document.querySelectorAll('[data-testid="add-card"]').forEach(btn => {
          btn.addEventListener('click', () => {
            const columnId = btn.closest('.column').dataset.testid.replace('column-', '');
            addNewCard(columnId);
          });
        });

        // Drag over column - allow drop
        Object.values(columns).forEach(column => {
          const contentArea = column.querySelector('.column-content');
          if (contentArea) {
            contentArea.addEventListener('dragover', (e) => {
              e.preventDefault();
              e.dataTransfer.dropEffect = 'move';
            });

            contentArea.addEventListener('drop', (e) => {
              e.preventDefault();
              if (!draggedCard) return;

              const toColumn = column.dataset.testid.replace('column-', '');

              // If dropping in the same column, do nothing
              if (draggedCard.fromColumn === toColumn) {
                draggedCard.element.classList.remove('opacity-50');
                return;
              }

              // Move the card
              moveCard(draggedCard.id, draggedCard.fromColumn, toColumn);

              // If moving to Done column, trigger confetti
              if (toColumn === 'done') {
                setTimeout(() => {
                  triggerConfetti();
                }, 100);
              }
            });
          }
        });

        // Delete modal buttons
        document.getElementById('confirm-delete').addEventListener('click', () => {
          if (cardToDelete) {
            deleteCard(cardToDelete.id);
            document.getElementById('delete-modal').classList.add('hidden');
          }
        });

        document.getElementById('cancel-delete').addEventListener('click', () => {
          document.getElementById('delete-modal').classList.add('hidden');
        });

        // Close modal when clicking backdrop
        document.getElementById('delete-modal').addEventListener('click', (e) => {
          if (e.target === document.getElementById('delete-modal')) {
            document.getElementById('delete-modal').classList.add('hidden');
          }
        });
      }

      // Add a new card to the specified column
      function addNewCard(columnId) {
        const contentArea = columns[columnId].querySelector('.column-content');

        // Create input element
        const input = document.createElement('textarea');
        input.dataset.testid = 'card-input';
        input.className = 'w-full p-3 border-2 border-blue-500 rounded-lg resize-none focus:ring-blue-500 focus:border-blue-600';
        input.placeholder = 'Enter card title...';
        input.rows = 2;
        input.style.maxHeight = '100px';

        // Insert at the beginning of the column
        contentArea.insertBefore(input, contentArea.firstChild);

        // Focus and select text
        input.focus();
        input.select();

        // Handle Enter key to save
        const handleKeyDown = (e) => {
          if (e.key === 'Enter' && !e.shiftKey) {
            e.preventDefault();
            saveNewCard(input, columnId);
          } else if (e.key === 'Escape') {
            e.preventDefault();
            cancelNewCard(input);
          }
        };

        // Handle blur to save
        const handleBlur = () => {
          saveNewCard(input, columnId);
        };

        input.addEventListener('keydown', handleKeyDown);
        input.addEventListener('blur', handleBlur);

        // Function to save new card
        function saveNewCard(input, columnId) {
          const title = input.value.trim();
          if (title === '') {
            cancelNewCard(input);
            return;
          }

          // Add the new card
          boardData[columnId].unshift({
            id: generateId(),
            title: title
          });

          saveData();
          renderBoard();

          // Remove event listeners
          input.removeEventListener('keydown', handleKeyDown);
          input.removeEventListener('blur', handleBlur);
        }

        // Function to cancel new card
        function cancelNewCard(input) {
          contentArea.removeChild(input);
          input.removeEventListener('keydown', handleKeyDown);
          input.removeEventListener('blur', handleBlur);
        }
      }

      // Move a card between columns
      function moveCard(cardId, fromColumn, toColumn) {
        // Find the card in the source column
        const cardIndex = boardData[fromColumn].findIndex(card => card.id === cardId);
        if (cardIndex === -1) return;

        const [movedCard] = boardData[fromColumn].splice(cardIndex, 1);

        // Add to the target column
        boardData[toColumn].unshift(movedCard);

        saveData();
        renderBoard();

        // Reset dragged card
        draggedCard = null;
      }

      // Delete a card
      function deleteCard(cardId) {
        columnIds.forEach(columnId => {
          const cardIndex = boardData[columnId].findIndex(card => card.id === cardId);
          if (cardIndex !== -1) {
            boardData[columnId].splice(cardIndex, 1);
            saveData();
            renderBoard();
          }
        });
      }

      // Trigger confetti animation
      function triggerConfetti() {
        const duration = 3000;
        const animationEnd = Date.now() + duration;
        const defaults = { startVelocity: 30, spread: 360, ticks: 120, zIndex: 0 };

        const randomInRange = (min, max) => Math.random() * (max - min) + min;

        const interval = setInterval(() => {
          const timeLeft = animationEnd - Date.now();

          if (timeLeft <= 0) {
            clearInterval(interval);
            return;
          }

          const particleCount = 50 * (timeLeft / duration);

          // since particles fall down, adjust for canvas height
          confetti({
            ...defaults,
            particleCount,
            origin: { x: randomInRange(0.1, 0.9), y: Math.random() - 0.2 }
          });
        }, 250);
      }
    </script>
  </div>
</body>
</html>

This implementation includes:

1. **Four Kanban Columns** with proper data-testid attributes
2. **Fully Functional Cards** that are:
   - Draggable across columns with smooth transitions
   - Editable in place (double-click or edit icon)
   - Persisted in localStorage with a custom namespace
   - Deletable with confirmation modal (no window.alert/prompt)
3. **Add Card Functionality** in each column with proper data-testid
4. **Tailwind CSS Styling** for a modern, vibrant design
5. **Confetti Animation** when cards move to Done column
6. **Dummy Data Generation** if no cards exist initially
7. **Comprehensive Comments** explaining the code structure
8. **All Required Test Selectors** for automated testing

The application maintains state across page reloads and provides a polished user experience with smooth animations and visual feedback.

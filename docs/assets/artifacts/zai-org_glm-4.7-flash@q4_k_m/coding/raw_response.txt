    <script>
        /**
         * Kanban Board Application Logic
         * Handles state management, local storage persistence, drag-and-drop,
         * and DOM manipulation.
         */

        // --- Configuration & State ---
        
        // Custom namespace for localStorage
        const STORAGE_KEY = 'kanban_board_v1_data';
        
        // Initial dummy data if storage is empty
        const INITIAL_DUMMY_DATA = {
            'backlog': [
                { id: 't1', title: 'Design System Setup' },
                { id: 't2', title: 'Research User Needs' }
            ],
            'in-progress': [
                { id: 't3', title: 'Implement Drag & Drop' }
            ],
            'review': [
                { id: 't4', title: 'Code Review: Auth Module' }
            ],
            'done': [
                { id: 't5', title: 'Project Setup' }
            ]
        };

        // Main application state
        let boardState = JSON.parse(JSON.stringify(INITIAL_DUMMY_DATA));
        
        // Track the ID of the card currently being deleted (for confirmation)
        let cardToDeleteId = null;

        // --- Initialization ---

        function init() {
            // Load from localStorage if available
            const stored = localStorage.getItem(STORAGE_KEY);
            if (stored) {
                boardState = JSON.parse(stored);
            } else {
                boardState = JSON.parse(JSON.stringify(INITIAL_DUMMY_DATA));
                saveToStorage();
            }
            renderBoard();
        }

        function saveToStorage() {
            localStorage.setItem(STORAGE_KEY, JSON.stringify(boardState));
        }

        // --- Rendering ---

        function renderBoard() {
            // Clear existing lists
            ['backlog', 'in-progress', 'review', 'done'].forEach(col => {
                const listEl = document.getElementById(`list-${col}`);
                listEl.innerHTML = '';
                
                // Update counter
                const countEl = document.getElementById(`count-${col}`);
                countEl.textContent = boardState[col].length;

                // Render cards
                boardState[col].forEach(card => {
                    const cardEl = createCardElement(card);
                    listEl.appendChild(cardEl);
                });
            });
        }

        function createCardElement(card) {
            const div = document.createElement('div');
            div.className = 'bg-white p-4 rounded-lg shadow-sm border border-slate-200 hover:shadow-md transition-shadow cursor-grab active:cursor-grabbing group animate-fade-in';
            div.draggable = true;
            div.setAttribute('data-testid', 'card');
            div.dataset.id = card.id;
            div.dataset.col = getColumnKeyFromId(card.id); // Helper to find current column

            // Content Editable Title
            const titleDiv = document.createElement('div');
            titleDiv.className = 'card-title text-slate-700 text-sm font-medium';
            titleDiv.contentEditable = true;
            titleDiv.textContent = card.title;
            
            // Event Listeners for Editing
            titleDiv.addEventListener('input', (e) => {
                // Update state on input (real-time)
                const col = div.dataset.col;
                const cardInState = boardState[col].find(c => c.id === card.id);
                if (cardInState) {
                    cardInState.title = e.target.textContent;
                    // We don't save to localStorage here to avoid performance hit on every keystroke
                }
            });

            titleDiv.addEventListener('blur', (e) => {
                // Save to localStorage on blur
                const col = div.dataset.col;
                const cardInState = boardState[col].find(c => c.id === card.id);
                if (cardInState) {
                    saveToStorage();
                }
            });

            // Prevent dropping content inside the card when dragging starts
            div.addEventListener('dragstart', (e) => {
                div.classList.add('dragging');
                e.dataTransfer.setData('text/plain', card.id);
                e.dataTransfer.effectAllowed = 'move';
            });

            div.addEventListener('dragend', () => {
                div.classList.remove('dragging');
                // Remove drag-over styles from all columns
                ['backlog', 'in-progress', 'review', 'done'].forEach(col => {
                    document.getElementById(`col-${col}`).classList.remove('drag-over');
                });
            });

            // Delete Button (Absolute positioned for better UX, or inline)
            const deleteBtn = document.createElement('button');
            deleteBtn.className = 'absolute top-2 right-2 text-slate-300 hover:text-red-500 opacity-0 group-hover:opacity-100 transition-opacity';
            deleteBtn.setAttribute('data-testid', 'delete-card');
            deleteBtn.innerHTML = `
                <svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                    <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16"></path>
                </svg>
            `;
            deleteBtn.onclick = () => openDeleteModal(card.id);

            div.appendChild(titleDiv);
            div.appendChild(deleteBtn);

            return div;
        }

        function getColumnKeyFromId(id) {
            for (const [key, cards] of Object.entries(boardState)) {
                if (cards.some(c => c.id === id)) return key;
            }
            return 'backlog';
        }

        // --- Drag and Drop Logic ---

        function handleDragOver(e) {
            e.preventDefault(); // Necessary to allow dropping
            e.dataTransfer.dropEffect = 'move';
            e.currentTarget.classList.add('drag-over');
        }

        function handleDragLeave(e) {
            e.currentTarget.classList.remove('drag-over');
        }

        function handleDrop(e, targetCol) {
            e.preventDefault();
            e.currentTarget.classList.remove('drag-over');

            const cardId = e.dataTransfer.getData('text/plain');
            if (!cardId) return;

            // Find the column the card is currently in
            const oldCol = getColumnKeyFromId(cardId);
            
            // If dropping in the same column, do nothing
            if (oldCol === targetCol) return;

            // Move the card in the state
            const cardIndex = boardState[oldCol].findIndex(c => c.id === cardId);
            if (cardIndex > -1) {
                const [card] = boardState[oldCol].splice(cardIndex, 1);
                boardState[targetCol].push(card);
                saveToStorage();
                renderBoard();

                // Trigger Confetti if moved to Done
                if (targetCol === 'done') {
                    triggerConfetti();
                }
            }
        }

        // --- Add Card Logic ---

        function showAddCardInput(colId) {
            const inputContainer = document.getElementById(`input-${colId}`);
            const input = inputContainer.querySelector('input');
            
            // Show container
            inputContainer.classList.remove('hidden');
            // Focus input
            input.focus();
            // Move cursor to end
            input.setSelectionRange(input.value.length, input.value.length);
        }

        function cancelAddCard(colId) {
            const inputContainer = document.getElementById(`input-${colId}`);
            const input = inputContainer.querySelector('input');
            input.value = ''; // Clear value
            inputContainer.classList.add('hidden');
        }

        function handleInputKeydown(e, colId) {
            if (e.key === 'Enter') {
                addCard(colId);
            }
        }

        function addCard(colId) {
            const inputContainer = document.getElementById(`input-${colId}`);
            const input = inputContainer.querySelector('input');
            const title = input.value.trim();

            if (title) {
                const newCard = {
                    id: 'task-' + Date.now(),
                    title: title
                };
                boardState[colId].push(newCard);
                saveToStorage();
                renderBoard();
                cancelAddCard(colId);
            }
        }

        // --- Delete Logic (Modal) ---

        function openDeleteModal(cardId) {
            cardToDeleteId = cardId;
            const modal = document.getElementById('delete-modal');
            modal.classList.remove('hidden');
        }

        function closeDeleteModal() {
            const modal = document.getElementById('delete-modal');
            modal.classList.add('hidden');
            cardToDeleteId = null;
        }

        function confirmDelete() {
            if (!cardToDeleteId) return;

            // Iterate through all columns to find and remove the card
            let found = false;
            for (const col in boardState) {
                const index = boardState[col].findIndex(c => c.id === cardToDeleteId);
                if (index > -1) {
                    boardState[col].splice(index, 1);
                    found = true;
                    break;
                }
            }

            if (found) {
                saveToStorage();
                renderBoard();
            }

            closeDeleteModal();
        }

        // --- Visual Effects ---

        function triggerConfetti() {
            // Simple confetti burst
            const count = 200;
            const defaults = {
                origin: { y: 0.7 },
                zIndex: 100
            };

            function fire(particleRatio, opts) {
                confetti(Object.assign({}, defaults, opts, {
                    particleCount: Math.floor(count * particleRatio)
                }));
            }

            fire(0.25, { spread: 26, startVelocity: 55, colors: ['#3b82f6', '#2563eb'] });
            fire(0.2, { spread: 60, colors: ['#60a5fa', '#3b82f6'] });
            fire(0.35, { spread: 100, decay: 0.91, scalar: 0.8 });
            fire(0.1, { spread: 120, startVelocity: 25, decay: 0.92, scalar: 1.2 });
            fire(0.1, { spread: 120, startVelocity: 45 });
        }

        // Start the app
        init();

    </script>
</body>

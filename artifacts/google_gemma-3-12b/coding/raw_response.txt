<body>

    <div class="container">

        <div data-testid="column-backlog" class="column">
            <h2>Backlog</h2>
            <div class="card-container"></div>
            <button data-testid="add-card" class="add-card">Add card</button>
        </div>

        <div data-testid="column-in-progress" class="column">
            <h2>In Progress</h2>
            <div class="card-container"></div>
            <button data-testid="add-card" class="add-card">Add card</button>
        </div>

        <div data-testid="column-review" class="column">
            <h2>Review</h2>
            <div class="card-container"></div>
            <button data-testid="add-card" class="add-card">Add card</button>
        </div>

        <div data-testid="column-done" class="column">
            <h2>Done</h2>
            <div class="card-container"></div>

    <script>
        const columnBacklog = document.querySelector('[data-testid="column-backlog"]');
        const columnInProgress = document.querySelector('[data-testid="column-in-progress"]');
        const columnReview = document.querySelector('[data-testid="column-review"]');
        const columnDone = document.querySelector('[data-testid="column-done"]');

        const cardContainerBacklog = columnBacklog.querySelector('.card-container');
        const cardContainerInProgress = columnInProgress.querySelector('.card-container');
        const cardContainerReview = columnReview.querySelector('.card-container');
        const cardContainerDone = columnDone.querySelector('.card-container');

        const deleteConfirmation = document.getElementById('deleteConfirmation');
        const confirmDeleteButton = document.querySelector('[data-testid="confirm-delete"]');
        const cancelDeleteButton = document.getElementById('cancelDelete');



        // Local Storage Namespace (to avoid conflicts)
        const localStorageNamespace = 'kanbanApp';

        // Function to load data from local storage
        function loadData() {
            const storedData = localStorage.getItem(localStorageNamespace);
            return storedData ? JSON.parse(storedData) : {
                backlog: [],
                inProgress: [],
                review: [],
                done: []
            };
        }

        // Function to save data to local storage
        function saveData(data) {
            localStorage.setItem(localStorageNamespace, JSON.stringify(data));
        }

        // Initial data from local storage or dummy cards if empty
        let appData = loadData();

        // Create some dummy cards if the backlog is empty
        if (appData.backlog.length === 0) {
            appData.backlog = [
                { id: '1', title: 'Design UI Mockups' },
                { id: '2', title: 'Implement User Authentication' }
            ];
        }

        // Function to render cards in a column
        function renderColumn(columnName, data) {
            const container = columnName === 'backlog' ? cardContainerBacklog :
                columnName === 'inProgress' ? cardContainerInProgress :
                    columnName === 'review' ? cardContainerReview : cardContainerDone;

            container.innerHTML = ''; // Clear existing cards

            data.forEach(card => {
                const cardElement = document.createElement('div');
                cardElement.classList.add('card');
                cardElement.setAttribute('data-testid', 'card');
                cardElement.dataset.id = card.id; // Store the ID

                cardElement.innerHTML = `
                    <div class="card-title">${card.title}</div>
                `;

                const deleteButton = document.createElement('button');
                deleteButton.classList.add('delete-card');
                deleteButton.setAttribute('data-testid', 'delete-card');
                deleteButton.textContent = 'Delete';

                cardElement.appendChild(deleteButton); // Add delete button to card element


                container.appendChild(cardElement);
            });
        }

        // Initial render of all columns
        renderColumn('backlog', appData.backlog);
        renderColumn('inProgress', appData.inProgress);
        renderColumn('review', appData.review);
        renderColumn('done', appData.done);

        // Function to create a new card
        function createCard(columnName) {
            const inputElement = document.createElement('textarea');
            inputElement.setAttribute('data-testid', 'card-input');
            inputElement.placeholder = 'Enter card title';

            // Wrap the textarea in a container to position it correctly
            const inputContainer = document.createElement('div');
            inputContainer.style.position = 'relative';

            // Add a small margin to the textarea
            inputElement.style.margin = '5px';

            inputContainer.appendChild(inputElement);


             // Add the input to the column
            const addCardButton = columnName === 'backlog' ? document.querySelector('[data-testid="add-card"]') :
                columnName === 'inProgress' ? document.querySelector('[data-testid="add-card"]') :
                    columnName === 'review' ? document.querySelector('[data-testid="add-card"]') : document.querySelector('[data-testid="add-card"]');

            const column = columnName === 'backlog' ? columnBacklog :
                columnName === 'inProgress' ? columnInProgress :
                    columnName === 'review' ? columnReview : columnDone;

            column.insertBefore(inputContainer, addCardButton);




            // Listen for Enter key press to commit the card
            inputElement.addEventListener('keydown', (event) => {
                if (event.key === 'Enter') {
                    const title = inputElement.value.trim();

                    if (title !== '') {
                        const newCard = { id: generateId(), title };

                        // Add card to the appropriate column's data
                        if (columnName === 'backlog') {
                            appData.backlog.push(newCard);
                        } else if (columnName === 'inProgress') {
                            appData.inProgress.push(newCard);
                        } else if (columnName === 'review') {
                            appData.review.push(newCard);
                        } else if (columnName === 'done') {
                            appData.done.push(newCard);
                        }

                        renderColumn(columnName, appData[columnName]); // Re-render the column
                        saveData(appData);

                        // Remove input and prompt for new card
                        inputContainer.remove(); // remove the container

                    }
                }
            });
        }



        // Function to generate a unique ID (simple implementation)
        function generateId() {
            return Math.random().toString(36).substring(2, 15) + Math.random().toString(36).substring(2, 15);
        }


        // Add card event listeners to each column
        document.querySelectorAll('[data-testid="add-card"]').forEach(button => {
            button.addEventListener('click', () => {
                const column = button.closest('.column');
                createCard(column.dataset.testid.slice(6));  // Extract column name from data-testid
            });
        });

        // Drag and drop functionality
        let draggedCard = null;

        document.querySelectorAll('.card').forEach(cardElement => {
            cardElement.addEventListener('dragstart', (event) => {
                draggedCard = cardElement;
                cardElement.classList.add('dragging'); // Add class for visual feedback
            });

            cardElement.addEventListener('dragend', () => {
                cardElement.classList.remove('dragging'); // Remove dragging class
                draggedCard = null;
            });
        });

        document.querySelectorAll('.column').forEach(columnElement => {
            columnElement.addEventListener('dragover', (event) => {
                event.preventDefault(); // Allow drop
            });

            columnElement.addEventListener('drop', (event) => {
                event.preventDefault(); // Prevent default behavior

                if (draggedCard) {
                    const cardId = draggedCard.dataset.id;

                    // Determine current and target columns
                    const fromColumnName = draggedCard.closest('.column').dataset.testid.slice(6);
                    const toColumnName = event.target.closest('.column').dataset.testid.slice(6);

                    // Remove card from the original column
                    const index = appData[fromColumnName].findIndex(card => card.id === cardId);
                    if (index !== -1) {
                        appData[fromColumnName].splice(index, 1);
                    }

                    // Add card to the target column
                    appData[toColumnName].push({ ...draggedCard.dataset, title: draggedCard.querySelector('.card-title').textContent });

                    // Re-render both columns
                    renderColumn(fromColumnName, appData[fromColumnName]);
                    renderColumn(toColumnName, appData[toColumnName]);

                    saveData(appData);

                    // Confetti Animation
                    if (toColumnName === 'done') {
                        triggerConfetti();
                    }

                }
            });
        });


        // Delete card event listener (confirmation)
        document.querySelectorAll('.delete-card').forEach(deleteButton => {
            deleteButton.addEventListener('click', (event) => {
                const cardElement = event.target.closest('.card');
                if (cardElement) {

                    const cardId = cardElement.dataset.id;
                     // Confirm deletion with prompt
                    deleteConfirmation.style.display = 'block'; // Show confirmation dialog

                }
            });
        });


         // Confirmation button click handler
         confirmDeleteButton.addEventListener('click', () => {

            const cardElement = draggedCard;
            if(cardElement){
                const cardId = cardElement.dataset.id;

                // Determine the current column name
                const fromColumnName = cardElement.closest('.column').dataset.testid.slice(6);

                // Remove the card from the data
                 const index = appData[fromColumnName].findIndex(card => card.id === cardId);
                if (index !== -1) {
                    appData[fromColumnName].splice(index, 1);

                }
                 renderColumn(fromColumnName, appData[fromColumnName]); // Re-render the column
                saveData(appData);

                draggedCard.remove(); // remove card from the DOM
                draggedCard = null;

                 deleteConfirmation.style.display = 'none'; // Hide confirmation dialog
            }


         });

        // Cancel button click handler
        cancelDeleteButton.addEventListener('click', () => {
            deleteConfirmation.style.display = 'none'; // Hide confirmation dialog
        });


        // Confetti animation trigger function
        function triggerConfetti() {
            const confettiContainer = document.querySelector('.confetti');

            for (let i = 0; i < 100; i++) {
                const confettiItem = document.createElement('span');
                confettiItem.classList.add('confetti-item');

                const randomColor = '#' + Math.floor(Math.random() * 16777215).toString(16);
                confettiItem.style.color = randomColor;

                const randomSize = Math.random() * 10 + 5;
                confettiItem.style.fontSize = randomSize + 'px';

                const animationDuration = Math.random() * 5 + 3;
                confettiItem.style.animationDuration = animationDuration + 's';

                confettiContainer.appendChild(confettiItem);
            }
        }


    </script>

</body>
</html>

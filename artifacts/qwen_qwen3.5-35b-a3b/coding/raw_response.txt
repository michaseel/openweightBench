    <div id="delete-modal" class="fixed inset-0 bg-black/50 z-50 hidden flex items-center justify-center backdrop-blur-sm transition-opacity duration-300">

            <h3 class="text-lg font-bold text-slate-800 mb-2">Delete Card?</h3>

    <script>
        /**
         * Kanban Board Application Logic
         * 
         * Namespace: KB (KanbanBoard) to avoid global pollution.
         */
        
        // --- State Management ---
        const STORAGE_KEY = 'kanban_board_state_v1';

        let state = {
            backlog: [],
            "in-progress": [],
            review: [],
            done: []
        };

        let draggedCardId = null;
        let currentDeleteTargetId = null; // ID of card to be deleted

        // --- Initialization ---

        function init() {
            loadState();
            
            // If empty, seed dummy data
            if (state.backlog.length === 0 && state["in-progress"].length === 0 && 
                state.review.length === 0 && state.done.length === 0) {
                seedDummyData();
            }

            renderBoard();
        }

        function loadState() {
            const stored = localStorage.getItem(STORAGE_KEY);
            if (stored) {
                try {
                    state = JSON.parse(stored);
                } catch (e) {
                    console.error("Failed to parse state", e);
                }
            }
        }

        function saveState() {
            localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
            updateCounts();
        }

        // --- Rendering ---

        function renderBoard() {
            ['backlog', 'in-progress', 'review', 'done'].forEach(columnId => {
                const container = document.querySelector(`[data-testid="column-${columnId}"] .flex.flex-col`);
                if (!container) return;

                container.innerHTML = ''; // Clear current cards
                
                state[columnId].forEach(card => {
                    const cardEl = createCardElement(card.id, card.title);
                    container.appendChild(cardEl);
                });
            });
            
            updateCounts();
        }

        function createCardElement(id, title) {
            const div = document.createElement('div');
            div.setAttribute('data-testid', 'card');
            div.className = "bg-white p-4 rounded-lg shadow-sm border border-slate-200 cursor-grab active:cursor-grabbing hover:shadow-md transition-shadow group relative animate-fade-in";
            div.draggable = true;
            
            // Drag Events
            div.addEventListener('dragstart', (e) => {
                draggedCardId = id;
                e.target.classList.add('dragging');
                e.dataTransfer.effectAllowed = 'move';
                // Store data for drop target to read
                e.dataTransfer.setData('text/plain', JSON.stringify({ id: id, sourceColumn: getCurrentColumnFromElement(e.target) })); 
            });

            div.addEventListener('dragend', (e) => {
                e.target.classList.remove('dragging');
                draggedCardId = null;
                
                // Remove drag-over styling from all columns
                document.querySelectorAll('.column').forEach(col => col.classList.remove('drag-over'));
            });

            // Editable Title
            const titleSpan = document.createElement('div');
            titleSpan.className = "editable text-slate-700 leading-snug cursor-text";
            titleSpan.contentEditable = true;
            titleSpan.innerText = title;
            
            // Save on blur (click away) or Enter key
            titleSpan.addEventListener('blur', () => updateCardTitle(id, titleSpan.innerText));
            titleSpan.addEventListener('keydown', (e) => {
                if(e.key === 'Enter') {
                    e.preventDefault();
                    titleSpan.blur();
                }
            });

            // Delete Button (Hidden by default, shown on hover/focus)
            const deleteBtn = document.createElement('button');
            deleteBtn.setAttribute('data-testid', 'delete-card');
            deleteBtn.className = "absolute top-2 right-2 p-1.5 text-slate-400 hover:text-red-600 hover:bg-red-50 rounded-md opacity-0 group-hover:opacity-100 transition-opacity focus-within:opacity-100";
            deleteBtn.title = "Delete card";
            
            // Stop propagation so clicking delete doesn't trigger drag start immediately if clicked while dragging logic is active (though unlikely)
            deleteBtn.addEventListener('click', (e) => {
                e.stopPropagation();
                openDeleteModal(id);
            });

            div.appendChild(titleSpan);
            div.appendChild(deleteBtn);

            return div;
        }

        function updateCounts() {
            ['backlog', 'in-progress', 'review', 'done'].forEach(col => {
                const countSpan = document.getElementById(`count-${col}`);
                if(countSpan) countSpan.innerText = state[col].length;
            });
        }

        // --- Actions: Add, Edit, Delete ---

        function focusInput(columnId) {
            const wrapper = document.getElementById('input-wrapper');
            const input = wrapper.querySelector('[data-testid="card-input"]');
            
            wrapper.classList.remove('hidden');
            input.value = '';
            input.focus();

            // Handle Enter key to submit new card
            input.addEventListener('keydown', (e) => {
                if (e.key === 'Enter' && !e.shiftKey) {
                    e.preventDefault();
                    commitNewCard(columnId, input.value.trim());
                }
            });
        }

        function commitNewCard(columnId, text) {
            if (!text.trim()) return;

            const newCard = {
                id: Date.now().toString(), // Simple ID generation
                title: text
            };

            state[columnId].unshift(newCard); // Add to top
            saveState();
            
            renderBoard();
            document.getElementById('input-wrapper').classList.add('hidden');
        }

        function updateCardTitle(id, newText) {
            if (!newText.trim()) return;
            
            // Find card in state and update
            for (let col of Object.keys(state)) {
                const index = state[col].findIndex(c => c.id === id);
                if (index !== -1) {
                    state[col][index].title = newText;
                    saveState();
                    return;
                }
            }
        }

        // --- Delete Logic ---

        function getCurrentColumnFromElement(el) {
            // Traverse up to find the column container
            const colContainer = el.closest('[data-testid^="column-"]');
            if(colContainer) {
                return colContainer.getAttribute('data-testid').replace('column-', '');
            }
            return null;
        }

        function openDeleteModal(id) {
            currentDeleteTargetId = id;
            const modal = document.getElementById('delete-modal');
            modal.classList.remove('hidden');
        }

        function closeDeleteModal() {
            document.getElementById('delete-modal').classList.add('hidden');
            currentDeleteTargetId = null;
        }

        // Bind the global button to remove modal
        document.getElementById('btn-confirm-delete').addEventListener('click', () => {
            if (!currentDeleteTargetId) return;

            for (let col of Object.keys(state)) {
                state[col] = state[col].filter(c => c.id !== currentDeleteTargetId);
                
                // Trigger Confetti if moved to Done (simulated by deleting from any and adding back? No, just delete). 
                // Wait, requirement says: "trigger a confetti animation when a card moves to 'Done'". 
                // This is usually done via Drag/Drop. I will handle that in the Drop handler below.
            }

            saveState();
            renderBoard();
            closeDeleteModal();
        });

        // --- Drag and Drop Logic ---

        function allowDrop(ev) {
            ev.preventDefault(); // Necessary to allow dropping
            
            const column = ev.target.closest('[data-testid^="column-"]');
            if (column && draggedCardId) {
                column.classList.add('drag-over');
            }
        }

        function handleDrop(ev) {
            ev.preventDefault();
            
            // Identify target column
            let dropTarget;
            if (ev.target.closest('[data-testid^="column-"]')) {
                dropTarget = ev.target.closest('[data-testid^="column-"]');
            } else if (ev.target.classList.contains('flex') && !ev.target.querySelector('.bg-white')) {
                 // Dropped on empty space of column container
                 dropTarget = ev.target; 
            }

            if (!dropTarget) return;

            const targetColumnId = dropTarget.getAttribute('data-testid').replace('column-', '');
            
            // Remove visual cues
            document.querySelectorAll('.column').forEach(c => c.classList.remove('drag-over'));

            if (!draggedCardId) return;

            // Find source column and card object
            let sourceColumnId = null;
            for (let col of Object.keys(state)) {
                const idx = state[col].findIndex(c => c.id === draggedCardId);
                if (idx !== -1) {
                    sourceColumnId = col;
                    break;
                }
            }

            // Move card in state
            if (sourceColumnId && targetColumnId) {
                const cardIndex = state[sourceColumnId].findIndex(c => c.id === draggedCardId);
                if (cardIndex > -1) {
                    const [card] = state[sourceColumnId].splice(cardIndex, 1);
                    state[targetColumnId].push(card); // Move to bottom of new column
                    
                    saveState();
                    renderBoard();

                    // Check for Confetti condition
                    if (targetColumnId === 'done') {
                        triggerConfetti();
                    }
                }
            }
        }

        // --- Confetti Animation ---

        function triggerConfetti() {
            const colors = ['#ef4444', '#f97316', '#eab308', '#22c55e', '#3b82f6', '#a855f7'];
            
            for (let i = 0; i < 50; i++) {
                const confetti = document.createElement('div');
                confetti.classList.add('confetti');
                
                // Randomize style
                const color = colors[Math.floor(Math.random() * colors.length)];
                const left = Math.random() * 100 + 'vw';
                const animDuration = (Math.random() * 2 + 2) + 's'; // 2-4s
                
                confetti.style.backgroundColor = color;
                confetti.style.left = left;
                confetti.style.top = '-10px';
                confetti.style.animationDuration = animDuration;

                document.body.appendChild(confetti);

                // Cleanup after animation
                setTimeout(() => {
                    confetti.remove();
                }, 4000);
            }
        }

        // Start App
        init();

    </script>
</body>

    <script>
        /**
         * APPLICATION STATE & CONFIGURATION
         */
        const STORAGE_KEY = 'vibrant_kanban_state_v1';
        let state = {
            backlog: [],
            inProgress: [],
            review: [],
            done: []
        };

        // Dummy data for first-time users
        const DUMMY_DATA = {
            backlog: [{ id: 'd1', text: 'Research new design trends' }, { id: 'd2', text: 'Write documentation' }],
            inProgress: [{ id: 'd3', text: 'Build Kanban logic' }],
            review: [{ id: 'd4', text: 'Code review for Auth module' }],
            done: [{ id: 'd5', text: 'Setup project structure' }]
        };

        /**
         * CORE STATE MANAGEMENT
         */
        function loadState() {
            const saved = localStorage.getItem(STORAGE_KEY);
            if (saved) {
                state = JSON.parse(saved);
            } else {
                state = JSON.parse(JSON.stringify(DUMMY_DATA));
                saveState();
            }
        }

        function saveState() {
            localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
        }

        /**
         * RENDERING LOGIC
         */
        function renderBoard() {
            const columns = ['backlog', 'inProgress', 'review', 'done'];
            
            columns.forEach(colKey => {
                const colId = `column-${colKey.replace(/[A-Z]/g, m => "-" + m.toLowerCase())}`;
                const container = document.getElementById(colId);
                if (!container) return;

                container.innerHTML = ''; // Clear current cards

                state[colKey].forEach(card => {
                    const cardEl = createCardElement(card, colKey);
                    container.appendChild(cardEl);
                });
            });
        }

        function createCardElement(card, colKey) {
            const div = document.createElement('div');
            div.setAttribute('data-testid', 'card');
            div.className = 'card-transition bg-white p-4 rounded-lg shadow-sm border border-slate-200 cursor-grab active:cursor-grabbing group relative hover:shadow-md hover:border-indigo-300';
            div.draggable = true;
            div.id = card.id;

            // Card text content
            const textSpan = document.createElement('p');
            textSpan.className = 'text-slate-700 font-medium break-words pr-6';
            textSpan.innerText = card.text;
            textSpan.onclick = () => openEditModal(card.id, colKey);

            // Delete button (hidden by default, shown on hover)
            const delBtn = document.createElement('button');
            delBtn.setAttribute('data-testid', 'delete-card');
            delBtn.className = 'absolute top-2 right-2 p-1 text-slate-300 hover:text-red-500 opacity-0 group-hover:opacity-100 transition-all';
            delBtn.innerHTML = `<svg xmlns="http://www.w3.org/2000/svg" class="h-4 w-4" fill="none" viewBox="0 0 24 24" stroke="currentColor"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12" /></svg>`;
            delBtn.onclick = (e) => {
                e.stopPropagation();
                openDeleteModal(card.id, colKey);
            };

            div.appendChild(textSpan);
            div.appendChild(delBtn);

            // Drag events
            div.ondragstart = (e) => {
                e.dataTransfer.setData('text/plain', JSON.stringify({ cardId: card.id, fromCol: colKey }));
                setTimeout(() => div.classList.add('dragging'), 0);
            };

            div.ondragend = () => div.classList.remove('dragging');

            return div;
        }

        /**
         * DRAG AND DROP HANDLERS
         */
        function initDragAndDrop() {
            const columns = document.querySelectorAll('[data-testid^="column-"]');

            columns.forEach(col => {
                col.ondragover = (e) => {
                    e.preventDefault();
                    col.classList.add('drag-over');
                };

                col.ondragleave = () => col.classList.remove('drag-over');

                col.ondrop = (e) => {
                    e.preventDefault();
                    col.classList.remove('drag-over');

                    const data = JSON.parse(e.dataTransfer.getData('text/plain'));
                    const targetColKey = col.id.replace('column-', '').replace(/-([a-z])/g, (g) => g[1].toUpperCase());
                    
                    moveCard(data.cardId, data.fromCol, targetColKey);
                };
            });
        }

        function moveCard(cardId, fromCol, toCol) {
            if (fromCol === toCol) return;

            // Find card in source and remove it
            const cardIdx = state[fromCol].findIndex(c => c.id === cardId);
            const [card] = state[fromCol].splice(cardIdx, 1);
            
            // Add to destination
            state[toCol].push(card);
            saveState();
            renderBoard();

            // Celebration if moved to Done
            if (toCol === 'done') {
                confetti({
                    particleCount: 100,
                    spread: 70,
                    origin: { y: 0.6 },
                    colors: ['#10b981', '#3b82f6', '#f59e0b']
                });
            }
        }

        /**
         * MODAL & INPUT LOGIC
         */
        let currentAction = { type: 'add', colKey: null, cardId: null };

        function openAddModal(colKey) {
            currentAction = { type: 'add', colKey: colKey, cardId: null };
            document.getElementById('modal-title').innerText = 'Add New Task';
            const input = document.getElementById('card-input');
            input.value = '';
            input.focus();
            document.getElementById('modal-overlay').classList.remove('hidden');
        }

        function openEditModal(cardId, colKey) {
            const card = state[colKey].find(c => c.id === cardId);
            currentAction = { type: 'edit', colKey: colKey, cardId: cardId };
            document.getElementById('modal-title').innerText = 'Edit Task';
            const input = document.getElementById('card-input');
            input.value = card.text;
            input.focus();
            document.getElementById('modal-overlay').classList.remove('hidden');
        }

        function closeModal() {
            document.getElementById('modal-overlay').classList.add('hidden');
        }

        function saveTaskFromModal() {
            const input = document.getElementById('card-input');
            const text = input.value.trim();
            if (!text) return;

            if (currentAction.type === 'add') {
                state[currentAction.colKey].push({
                    id: 'c' + Date.now(),
                    text: text
                });
            } else {
                const card = state[currentAction.colKey].find(c => c.id === currentAction.cardId);
                if (card) card.text = text;
            }

            saveState();
            renderBoard();
            closeModal();
        }

        // Delete Modal Logic
        let deleteTarget = { id: null, colKey: null };
        function openDeleteModal(cardId, colKey) {
            deleteTarget = { id: cardId, colKey: colKey };
            document.getElementById('delete-modal').classList.remove('hidden');
        }

        function closeDeleteModal() {
            document.getElementById('delete-modal').classList.add('hidden');
        }

        function confirmDeleteTask() {
            const { id, colKey } = deleteTarget;
            state[colKey] = state[colKey].filter(c => c.id !== id);
            saveState();
            renderBoard();
            closeDeleteModal();
        }

        /**
         * EVENT LISTENERS
         */
        function initEventListeners() {
            // Save button in modal
            document.getElementById('modal-save-btn').onclick = saveTaskFromModal;

            // Enter key to commit new card/edit
            document.getElementById('card-input').onkeydown = (e) => {
                if (e.key === 'Enter' && !e.shiftKey) {
                    e.preventDefault();
                    saveTaskFromModal();
                }
            };

            // Confirm delete button
            document.getElementById('modal-confirm-delete').onclick = confirmDeleteTask;

            // Close modals on backdrop click
            window.onclick = (e) => {
                if (e.target.id === 'modal-overlay') closeModal();
                if (e.target.id === 'delete-modal') closeDeleteModal();
            };
        }

        /**
         * INITIALIZATION
         */
        (function init() {
            loadState();
            renderBoard();
            initDragAndDrop();
            initEventListeners();
        })();
    </script>

     * Load board state from localStorage.

     */
    function loadState() {
      try {
        const raw = localStorage.getItem(STORAGE_KEY);
        if (raw) {
          const parsed = JSON.parse(raw);
          // Validate: must be an array
          if (Array.isArray(parsed)) {

      } catch (e) {
        console.warn('Failed to load state from localStorage:', e);
      }
      // No valid state found — return defaults

    }

      actionsRow.appendChild(deleteBtn);

      cardEl.appendChild(actionsRow);

      return cardEl;

    /**
     * Format a timestamp into a human-readable relative string.
     */
    function formatTime(timestamp) {
      const diff = Date.now() - timestamp;
      const mins = Math.floor(diff / 60000);
      if (mins < 1) return 'just now';
      if (mins < 60) return `${mins}m ago`;
      const hours = Math.floor(mins / 60);
      if (hours < 24) return `${hours}h ago`;
      const days = Math.floor(hours / 24);
      if (days < 7) return `${days}d ago`;
      return new Date(timestamp).toLocaleDateString();
    }

    // ===== 5. DRAG & DROP (HTML5 DnD API) =====

    /** Currently dragged card element (for visual feedback) */
    let draggedCardEl = null;

    /**
     * Set up drag-and-drop event listeners on the board.
     * Uses event delegation for efficiency.
     */
    function setupDragAndDrop() {
      // Drag start: mark the card being dragged
      boardEl.addEventListener('dragstart', (e) => {
        const cardEl = e.target.closest('[data-testid="card"]');
        if (!cardEl) return;

        draggedCardEl = cardEl;
        const cardId = cardEl.getAttribute('data-card-id');
        e.dataTransfer.setData('text/plain', cardId);
        e.dataTransfer.effectAllowed = 'move';

        // Add visual feedback class after a brief delay (so the drag image renders first)
        requestAnimationFrame(() => {
          cardEl.classList.add('dragging');
        });
      });

      // Drag end: clean up visual state
      boardEl.addEventListener('dragend', (e) => {
        if (draggedCardEl) {
          draggedCardEl.classList.remove('dragging');
          draggedCardEl = null;
        }
        // Remove all drag-over highlights
        document.querySelectorAll('.column-drop-zone').forEach(zone => {
          zone.classList.remove('drag-over');
        });
      });

      // Drag over: allow drop and highlight the target column
      boardEl.addEventListener('dragover', (e) => {
        e.preventDefault();
        e.dataTransfer.dropEffect = 'move';

        const dropZone = e.target.closest('.column-drop-zone');
        if (dropZone) {
          // Remove highlight from all zones
          document.querySelectorAll('.column-drop-zone').forEach(zone => {
            zone.classList.remove('drag-over');
          });
          // Highlight the current zone
          dropZone.classList.add('drag-over');
        }
      });

      // Drag leave: remove highlight
      boardEl.addEventListener('dragleave', (e) => {
        const dropZone = e.target.closest('.column-drop-zone');
        if (dropZone && !dropZone.contains(e.relatedTarget)) {
          dropZone.classList.remove('drag-over');
        }
      });

      // Drop: move the card to the new column
      boardEl.addEventListener('drop', (e) => {
        e.preventDefault();

        const dropZone = e.target.closest('.column-drop-zone');
        if (!dropZone) return;

        const cardId = e.dataTransfer.getData('text/plain');
        const newColumnId = dropZone.getAttribute('data-column-id');

        if (!cardId || !newColumnId) return;

        // Find the card in state
        const cardIndex = cards.findIndex(c => c.id === cardId);
        if (cardIndex === -1) return;

        const oldColumnId = cards[cardIndex].columnId;

        // Update the card's column
        cards[cardIndex].columnId = newColumnId;

        // Save state
        saveState(cards);

        // Re-render the board
        renderBoard();

        // Remove drag-over highlights
        document.querySelectorAll('.column-drop-zone').forEach(zone => {
          zone.classList.remove('drag-over');
        });

        // Trigger confetti if card was moved to "Done"
        if (newColumnId === 'done' && oldColumnId !== 'done') {
          triggerConfetti();
        }
      });
    }

    // ===== 6. CARD CRUD OPERATIONS =====

    /**
     * Set up event delegation for card actions (add, edit, delete).
     */
    function setupCardActions() {
      // --- Add Card Button Click ---
      boardEl.addEventListener('click', (e) => {
        const addBtn = e.target.closest('[data-testid="add-card"]');
        if (addBtn) {
          const columnId = addBtn.getAttribute('data-column-id');
          showAddCardInput(columnId, addBtn);
          return;
        }

        // --- Edit Card Button Click ---
        const editBtn = e.target.closest('[data-action="edit"]');
        if (editBtn) {
          const cardEl = editBtn.closest('[data-testid="card"]');
          if (cardEl) {
            startEditingCard(cardEl);
          }
          return;
        }

        // --- Delete Card Button Click ---
        const deleteBtn = e.target.closest('[data-testid="delete-card"]');
        if (deleteBtn) {
          const cardEl = deleteBtn.closest('[data-testid="card"]');
          if (cardEl) {
            const cardId = cardEl.getAttribute('data-card-id');
            showDeleteConfirmation(cardId);
          }
          return;
        }
      });

      // --- Confirm Delete ---
      confirmDeleteBtn.addEventListener('click', () => {
        if (pendingDeleteCardId) {
          deleteCard(pendingDeleteCardId);
          hideDeleteModal();
        }
      });

      // --- Cancel Delete ---
      cancelDeleteBtn.addEventListener('click', () => {
        hideDeleteModal();
      });

      // --- Close modal on overlay click ---
      deleteModal.addEventListener('click', (e) => {
        if (e.target === deleteModal) {
          hideDeleteModal();
        }
      });

      // --- Close modal on Escape key ---
      document.addEventListener('keydown', (e) => {
        if (e.key === 'Escape' && !deleteModal.classList.contains('hidden')) {
          hideDeleteModal();
        }
      });

      // --- Reset Board ---
      resetBtn.addEventListener('click', () => {
        if (confirm('Reset the board to default cards? All changes will be lost.')) {
          cards = DEFAULT_CARDS.map(c => ({ ...c }));
          saveState(cards);
          renderBoard();
        }
      });
    }

    /**
     * Show an inline input to add a new card to a column.
     * Replaces the "Add card" button temporarily with an input field.
     */
    function showAddCardInput(columnId, addBtn) {
      const column = COLUMNS.find(c => c.id === columnId);
      if (!column) return;

      // Find the drop zone for this column
      const dropZone = document.querySelector(`[data-testid="${column.testId}"]`);
      if (!dropZone) return;

      // Hide the add button
      addBtn.style.display = 'none';

      // Create the input wrapper
      const inputWrapper = document.createElement('div');
      inputWrapper.className = 'add-card-input-wrapper';

      const input = document.createElement('input');
      input.setAttribute('data-testid', 'card-input');
      input.className = 'card-edit-input w-full';
      input.type = 'text';
      input.placeholder = 'Enter card title...';
      input.maxLength = 200;

      inputWrapper.appendChild(input);
      dropZone.appendChild(inputWrapper);

      // Focus the input
      input.focus();

      /**
       * Commit the new card: create it, save state, re-render.
       */
      function commitCard() {
        const title = input.value.trim();
        if (!title) {
          // If empty, just clean up and show the button again
          cleanup();
          return;
        }

        // Create new card object
        const newCard = {
          id: generateId(),
          columnId: columnId,
          title: title,
          createdAt: Date.now(),
        };

        // Add to state
        cards.push(newCard);
        saveState(cards);

        // Re-render
        renderBoard();
      }

      /**
       * Clean up the input UI and restore the add button.
       */
      function cleanup() {
        inputWrapper.remove();
        addBtn.style.display = '';
      }

      // Enter key commits the card
      input.addEventListener('keydown', (e) => {
        if (e.key === 'Enter') {
          e.preventDefault();
          commitCard();
        } else if (e.key === 'Escape') {
          cleanup();
        }
      });

      // Blur: if the input still has focus when blurred, commit
      input.addEventListener('blur', () => {
        // Small delay to allow click events to process first
        setTimeout(() => {
          if (document.activeElement === input) {
            commitCard();
          } else if (inputWrapper.parentNode) {
            cleanup();
          }
        }, 100);
      });
    }

    /**
     * Start editing a card in place: replace the title text with an input.
     */
    function startEditingCard(cardEl) {
      const cardId = cardEl.getAttribute('data-card-id');
      const card = cards.find(c => c.id === cardId);
      if (!card) return;

      // Find the title element
      const titleEl = cardEl.querySelector('.card-title');
      if (!titleEl) return;

      // Replace title with an input
      const input = document.createElement('input');
      input.className = 'card-edit-input w-full';
      input.value = card.title;
      input.maxLength = 200;

      titleEl.replaceWith(input);
      input.focus();
      input.select();

      /**
       * Save the edited card title.
       */
      function saveEdit() {
        const newTitle = input.value.trim();
        if (newTitle) {
          card.title = newTitle;
          saveState(cards);
        }
        renderBoard();
      }

      // Enter saves
      input.addEventListener('keydown', (e) => {
        if (e.key === 'Enter') {
          e.preventDefault();
          saveEdit();
        } else if (e.key === 'Escape') {
          renderBoard(); // Cancel: re-render to restore original
        }
      });

      // Blur saves
      input.addEventListener('blur', () => {
        setTimeout(() => {
          if (document.activeElement === input) {
            saveEdit();
          }
        }, 100);
      });
    }

    /**
     * Show the delete confirmation modal for a given card ID.
     */
    function showDeleteConfirmation(cardId) {
      const card = cards.find(c => c.id === cardId);
      if (!card) return;

      pendingDeleteCardId = cardId;
      deleteModalTitle.textContent = `"${card.title}"`;
      deleteModal.classList.remove('hidden');
    }

    /**
     * Hide the delete confirmation modal.
     */
    function hideDeleteModal() {
      deleteModal.classList.add('hidden');
      pendingDeleteCardId = null;
    }

    /**
     * Delete a card from state and re-render.
     */
    function deleteCard(cardId) {
      cards = cards.filter(c => c.id !== cardId);
      saveState(cards);
      renderBoard();
    }

    // ===== 7. CONFETTI ANIMATION =====

    /**
     * Simple canvas-based confetti particle system.
     * Triggered when a card is moved to the "Done" column.
     */
    function triggerConfetti() {
      const canvas = document.getElementById('confetti-canvas');
      const ctx = canvas.getContext('2d');

      // Set canvas to full window size
      canvas.width = window.innerWidth;
      canvas.height = window.innerHeight;

      // Particle colors — vibrant palette
      const colors = [
        '#8b5cf6', '#a78bfa', '#c084fc', '#e879f9',
        '#f472b6', '#fb7185', '#f87171', '#fb923c',
        '#fbbf24', '#a3e635', '#34d399', '#22d3ee',
        '#60a5fa', '#f0abfc', '#fda4af',
      ];

      /**
       * Particle object representing a single confetti piece.
       */
      class Particle {
        constructor() {
          this.x = Math.random() * canvas.width;
          this.y = -10;
          this.size = Math.random() * 8 + 4;
          this.color = colors[Math.floor(Math.random() * colors.length)];
          this.speedY = Math.random() * 3 + 2;
          this.speedX = (Math.random() - 0.5) * 4;
          this.rotation = Math.random() * 360;
          this.rotationSpeed = (Math.random() - 0.5) * 10;
          this.opacity = 1;
          this.decay = Math.random() * 0.008 + 0.005;
          // Shape: 0 = rectangle, 1 = circle
          this.shape = Math.random() > 0.5 ? 0 : 1;
          // Wobble for organic feel
          this.wobble = Math.random() * Math.PI * 2;
          this.wobbleSpeed = Math.random() * 0.05 + 0.02;
        }

        update() {
          this.y += this.speedY;
          this.x += this.speedX + Math.sin(this.wobble) * 0.5;
          this.wobble += this.wobbleSpeed;
          this.rotation += this.rotationSpeed;
          this.opacity -= this.decay;
          // Slight gravity acceleration
          this.speedY += 0.05;
        }

        draw(ctx) {
          ctx.save();
          ctx.translate(this.x, this.y);
          ctx.rotate((this.rotation * Math.PI) / 180);
          ctx.globalAlpha = Math.max(0, this.opacity);
          ctx.fillStyle = this.color;

          if (this.shape === 0) {
            // Rectangle confetti
            ctx.fillRect(-this.size / 2, -this.size / 4, this.size, this.size / 2);
          } else {
            // Circle confetti
            ctx.beginPath();
            ctx.arc(0, 0, this.size / 2, 0, Math.PI * 2);
            ctx.fill();
          }

          ctx.restore();
        }
      }

      // Create particles
      const particles = [];
      const particleCount = 150;

      for (let i = 0; i < particleCount; i++) {
        // Stagger the creation for a more natural burst
        setTimeout(() => {
          particles.push(new Particle());
        }, Math.random() * 500);
      }

      /**
       * Animation loop: update and draw all particles.
       */
      function animate() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);

        // Filter out dead particles
        const alive = particles.filter(p => p.opacity > 0 && p.y < canvas.height + 20);

        if (alive.length === 0) {
          // Animation complete — clean up
          ctx.clearRect(0, 0, canvas.width, canvas.height);
          return;
        }

        alive.forEach(p => {
          p.update();
          p.draw(ctx);
        });

        requestAnimationFrame(animate);
      }

      // Start animation
      animate();
    }

    // ===== 8. UTILITY FUNCTIONS =====

    /**
     * Generate a unique ID for a new card.
     * Uses a timestamp + random string for uniqueness.
     */
    function generateId() {
      return 'card-' + Date.now().toString(36) + '-' + Math.random().toString(36).substring(2, 8);
    }

    /**
     * Initialize the application:
     * 1. Render the board with current state
     * 2. Set up drag-and-drop
     * 3. Set up card action event listeners
     */
    function init() {
      renderBoard();
      setupDragAndDrop();
      setupCardActions();
    }

    // Run on DOM ready
    document.addEventListener('DOMContentLoaded', init);
  </script>

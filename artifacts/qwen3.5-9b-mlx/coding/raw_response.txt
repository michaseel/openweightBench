    <script>
        // --- State Management ---
        const STORAGE_KEY = 'kanban-board-v1';

        // Initial Dummy Data
        const initialData = [
            { id: '1', title: 'Research Competitors', columnId: 'backlog' },
            { id: '2', title: 'Design System Draft', columnId: 'backlog' },
            { id: '3', title: 'Setup Database Schema', columnId: 'in-progress' },
            { id: '4', title: 'Implement Login API', columnId: 'in-progress' },
            { id: '5', title: 'User Testing Session 1', columnId: 'review' },
            { id: '6', title: 'Fix Navigation Bug', columnId: 'review' },
            { id: '7', title: 'Deploy to Staging', columnId: 'done' },
        ];

        // Load State or Initialize
        let appState = {};

        function loadState() {
            const stored = localStorage.getItem(STORAGE_KEY);
            if (stored) {
                appState = JSON.parse(stored);
            } else {
                // Initialize columns if empty
                appState = {
                    'backlog': [],
                    'in-progress': [],
                    'review': [],
                    'done': []
                };
                // Add dummy data if empty state
                initialData.forEach(card => appState[card.columnId].push(card));
            }
        }

        function saveState() {
            localStorage.setItem(STORAGE_KEY, JSON.stringify(appState));
        }

        // --- Rendering Logic ---
        
        function renderBoard() {
            const columns = ['backlog', 'in-progress', 'review', 'done'];
            
            columns.forEach(colId => {
                const container = document.getElementById(`list-${colId}`);
                if (!container) return;

                container.innerHTML = ''; // Clear current column
                
                const cardsInColumn = appState[colId] || [];

                cardsInColumn.forEach((card, index) => {
                    const cardEl = document.createElement('div');
                    cardEl.className = 'kanban-card relative group';
                    cardEl.setAttribute('data-testid', 'card');
                    cardEl.dataset.id = card.id;
                    cardEl.draggable = true;

                    // Card HTML Structure: Title + Delete Button (hidden by default or visible?)
                    // Requirement: Editable in place. Let's use contenteditable for simplicity 
                    // BUT requirement says "input/textarea" with testid.
                    
                    /* Implementation Detail: 
                       To satisfy `data-testid="card-input"` for editing, we need an input.
                       However, "Editable in place" usually means clicking text makes it editable.
                       We will use a hybrid: 
                       Default view is <span>. Clicking it reveals an <input> with the testid.
                    */

                    // Title Wrapper (Clickable to Edit)
                    const titleWrapper = document.createElement('div');
                    
                    // The Input (Hidden by default)
                    const inputEl = document.createElement('input');
                    inputEl.type = 'text';
                    inputEl.value = card.title;
                    inputEl.className = `bg-transparent border-none outline-none px-1 py-0.5 text-gray-800 font-medium focus:ring-2 rounded transition-all duration-200 w-full`;
                    inputEl.setAttribute('data-testid', 'card-input'); // Mandatory Test ID
                    inputEl.style.display = 'none';

                    const titleSpan = document.createElement('span');
                    titleSpan.textContent = card.title;
                    titleSpan.className = "text-gray-800 font-medium text-lg cursor-text select-none";
                    titleSpan.dataset.editable = "true";

                    // Click to Edit Logic
                    const isEditing = card.isEditing;
                    if (isEditing) {
                        inputEl.style.display = 'block';
                        titleSpan.style.display = 'none';
                        // Auto focus when switching modes or just on load if editing? 
                        // Let's auto-focus
                        inputEl.focus();
                    } else {
                        titleSpan.style.display = 'block';
                        // Click span to toggle edit
                        const clickHandler = (e) => {
                            if(e.target === titleSpan && !inputEl.isContentEditable) { // Check not already in input
                                toggleEdit(card.id);
                            }
                        };
                        titleSpan.addEventListener('click', clickHandler);
                    }

                    // Delete Button (Trash Icon)
                    const deleteBtn = document.createElement('button');
                    deleteBtn.className = `absolute top-2 right-2 p-1 rounded hover:bg-red-50 text-gray-400 hover:text-red-500 transition-opacity opacity-0 group-hover:opacity-100`;
                    deleteBtn.innerHTML = '&times;'; // Simple X
                    deleteBtn.setAttribute('data-testid', 'delete-card'); // Mandatory Test ID
                    deleteBtn.onclick = () => handleDelete(card.id);

                    titleWrapper.appendChild(titleSpan);
                    titleWrapper.appendChild(inputEl);
                    
                    cardEl.appendChild(titleWrapper);
                    cardEl.appendChild(deleteBtn);

                    // Drag Events
                    setupDragEvents(cardEl, colId);

                    container.appendChild(cardEl);
                });
            });
        }

        // --- Drag and Drop Logic (HTML5 API) ---
        
        let draggedCardId = null;
        let sourceColumnId = null;

        function setupDragEvents(el, colId) {
            el.addEventListener('dragstart', handleDragStart);
            
            // Add a "ghost" drop zone logic. 
            // We attach listeners to the column containers or specific items inside?
            // Better UX: Attach listeners to the container divs.
        }

        document.querySelectorAll('[data-testid="column-"]').forEach(col => {
            col.addEventListener('dragover', (e) => e.preventDefault()); // Allow drop
        });

        document.querySelectorAll('.kanban-column').forEach(col => {
            col.addEventListener('dragover', (e) => {
                // Highlighting drop zone could be added here via CSS class
            });

            col.addEventListener('drop', (e) => {
                e.preventDefault();
                if (!draggedCardId) return;

                const targetColId = col.id.replace('column-', '');
                
                // Move item in state
                moveCard(draggedCardId, sourceColumnId, targetColId);
                
                // Re-render just that column to save performance? 
                // Or full re-render. For simplicity in vanilla JS, we'll trigger a check.
            });
        });

        function handleDragStart(e) {
            draggedCardId = this.dataset.id;
            sourceColumnId = this.parentElement.id.replace('list-', ''); // Get column from parent list
            
            // Add visual class
            this.classList.add('dragging');
        }

        function handleDragEnd(e) {
            this.classList.remove('dragging');
            draggedCardId = null;
        }

        function handleDelete(id) {
            // Show Modal
            const modal = document.getElementById('delete-modal');
            const confirmBtn = document.getElementById('confirm-delete-btn');
            
            // Populate modal title with card name if needed, currently generic
            confirmBtn.onclick = () => {
                // 1. Remove from state
                const colId = appState[id.columnId];
                const index = colId.findIndex(c => c.id === id);
                
                if (index > -1) {
                    // Remove item from array
                    colId.splice(index, 1);
                    
                    // Check if empty to remove key (optional cleanup)
                    if(colId.length === 0) delete appState[id.columnId];

                    saveState();
                    
                    // Close Modal
                    modal.classList.add('hidden');
                    
                    // Trigger Confetti if moving to Done or just deleting? 
                    // Requirement: "trigger a confetti animation when a card moves to Done"
                    
                }
            };

            // Show modal & wait for confirm
            modal.classList.remove('hidden');
        }

        function moveCard(id, fromColId, toColId) {
            // Remove from source
            const sourceCards = appState[fromColId];
            const cardIndex = sourceCards.findIndex(c => c.id === id);
            
            if (cardIndex !== -1) {
                sourceCards.splice(cardIndex, 1); // Remove from array

                if (!appState[toColId]) {
                    appState[toColId] = [];
                }

                // Add to target (Append logic for simplicity, or insert at specific index?)
                // Standard DnD often appends to end of target unless dropped between items.
                // For this implementation, we will append to the target column.
                appState[toColId].push(sourceCards[cardIndex]); // Add the removed card object
                
                saveState();
            }
        }

        function toggleEdit(id) {
            // In state, mark as editing? Or just render differently.
            // Since we re-render, let's add an `isEditing` flag to the card object in state
            // Find the card first. We need a way to track which ID is being edited currently globally?
            // Actually, simpler: Toggle the state of that specific card.
            
            const cards = appState['backlog'].concat(appState['in-progress']).concat(appState['review']).concat(appState['done']);
            const card = cards.find(c => c.id === id);
            
            if(card) {
                // Toggle a property in state
                card.isEditing = !card.isEditing; 
                saveState(); // Save the toggle state
            } else {
                console.error("Card not found for edit");
            }
        }

        function saveNewCard() {
            const input = document.getElementById('card-input');
            const title = input.value.trim();
            
            if (!title) {
                alert("Title cannot be empty"); // Fallback for validation logic
                return;
            }

            const currentColumn = document.querySelector('.modal.active') ? getActiveColumnFromModal() : 'backlog';
            
            // Generate unique ID
            const newId = Date.now().toString() + Math.random().toString(36).substr(2, 9);

            const newCard = {
                id: newId,
                title: title,
                columnId: currentColumn || 'backlog', // Default to backlog if modal doesn't have active col
                isEditing: false
            };

            // Add to state
            if (!appState[currentColumn]) {
                appState[currentColumn] = []; // Ensure column exists in state
            }
            appState[currentColumn].push(newCard);

            saveState();
            
            // Close Modal & Re-render
            closeAddModal();
            renderBoard();

            // Trigger Confetti (as it's new/done-like)
            fireConfetti();
        }

        // Helper to track which column the modal belongs to (for adding)
        let activeModalColumn = 'backlog';

        function openAddModal(colId) {
            activeModalColumn = colId;
            
            const modal = document.getElementById('add-card-modal');
            const input = document.getElementById('card-input');

            // Show modal
            modal.classList.remove('hidden');
            
            // Focus input & wait for Enter key (Requirement)
            setTimeout(() => {
                input.focus();
                // Add listener for Enter key
                const onKeyDown = (e) => {
                    if(e.key === 'Enter') {
                        saveNewCard(); // This will close modal on success
                    } else if (e.key === 'Escape') {
                        closeAddModal();
                    }
                };
                
                input.addEventListener('keydown', onKeyDown);

                // Cleanup listener after user types (timeout)
                setTimeout(() => {
                    input.removeEventListener('keydown', onKeyDown);
                }, 1000);

            }, 50);
        }

        function closeAddModal() {
            const modal = document.getElementById('add-card-modal');
            modal.classList.add('hidden');
        }

        // --- Confetti Logic (Vanilla JS) ---
        
        function fireConfetti() {
            const container = document.getElementById('confetti-canvas');
            container.classList.remove('hidden');

            const colors = ['#ef4444', '#f97316', '#eab308', '#22c55e'];
            const particleCount = 50;

            for (let i = 0; i < particleCount; i++) {
                const piece = document.createElement('div');
                piece.classList.add('confetti-piece');
                
                // Random properties
                const bg = colors[Math.floor(Math.random() * colors.length)];
                piece.style.backgroundColor = bg;
                
                // Start at center screen (or column location) - let's do center for "Done" celebration
                // Actually, if it moves to Done, maybe spawn near the Done column. 
                // Let's just center for now as it's a global celebration.
                
                const startX = Math.random() * window.innerWidth;
                const startY = Math.random() * (window.innerHeight - 100); // Keep away from top nav
                
                piece.style.left = startX + 'px';
                piece.style.top = startY + 'px';

                container.appendChild(piece);

                // Animate
                const duration = 1000 + Math.random() * 500; // ms
                const delay = Math.random() * 200;

                piece.animate([
                    { transform: 'translate(0, 0) scale(1)', opacity: 1 },
                    { transform: `translate(${Math.random() * 200 - 100}px, ${window.innerHeight + 50}px) scale(0.2)`, opacity: 0 }
                ], {
                    duration: duration,
                    easing: 'cubic-bezier(0.25, 1, 0.5, 1)',
                    delay: delay
                });

                // Remove from DOM after animation
                setTimeout(() => {
                    piece.remove();
                }, duration + delay);

                // Random rotation
                piece.style.transform = `rotate(${Math.random() * 360}deg)`;
            }

            setTimeout(() => {
                container.classList.add('hidden');
            }, 1500); // Keep for a bit then clean up DOM
        }

        function getActiveColumnFromModal() {
            // Find the button that triggered the modal or check which column was active when clicked?
            // Since we don't store "who called me", let's look at the DOM structure if possible,
            // or simpler: Just default to 'backlog' if we can't determine. 
            // Better approach: Pass column ID in openAddModal (already done).
            return activeModalColumn; 
        }

        // --- Initialization ---
        
        window.addEventListener('DOMContentLoaded', () => {
            loadState();
            
            // Re-render initial columns or existing state
            renderBoard();

            // Set up the "Enter" key listener for inputs globally (for new cards)
            document.addEventListener('keydown', (e) => {
                if (document.activeElement.id === 'card-input') {
                    saveNewCard();
                }
            });

            // Re-apply Drag Events to existing DOM elements after render (since we rebuild DOM)
            document.querySelectorAll('.kanban-card').forEach(card => {
                // Re-attach listeners manually or just let the render function handle it? 
                // We must re-bind them because elements are recreated.
                
                card.addEventListener('dragstart', handleDragStart);
                card.addEventListener('dragend', handleDragEnd);

                // Re-bind click handlers for edit/delete
                card.querySelectorAll('[data-testid="delete-card"]').forEach(btn => {
                    btn.onclick = () => handleDelete(card.dataset.id);
                });

                // Re-bind click for title (edit) - this is tricky because we need to find the span vs input
                // We'll rely on the render logic for initial setup, but we need to attach listeners 
                // inside the click handler or re-run toggle logic.
            });

            // Re-attach Listeners for Add Buttons (since we rebuild DOM)
            document.querySelectorAll('[data-testid="add-card"]').forEach(btn => {
                btn.onclick = (e) => {
                    e.preventDefault(); // prevent button click propagation if needed
                    openAddModal(e.target.closest('.flex-col').id.replace('column-', ''));
                };
            });

            // Re-attach Confirm Delete Button (since we rebuild DOM)
            document.getElementById('confirm-delete-btn').onclick = () => {
                // We need to know WHICH card is being deleted. 
                // The modal doesn't know the ID unless we store it in a global var.
            };

            // Fix Delete Modal Logic (Hardcoded for simplicity in this structure):
            const modal = document.getElementById('delete-modal');
            const confirmBtn = document.getElementById('confirm-delete-btn');

            // We need a global reference to the card being deleted when modal opens
            window.currentDeleteId = null;

            // Redefine handleDelete to capture ID in a global var
            const originalHandleDelete = window.handleDelete;
            
            // Re-define handleDelete for the global scope used by the button
            window.handleDelete = function(id) {
                window.currentDeleteId = id;

                const modal = document.getElementById('delete-modal');
                confirmBtn.onclick = () => {
                    // Remove from state
                    const colId = appState[id.columnId];
                    const index = colId.findIndex(c => c.id === id);
                    
                    if (index > -1) {
                        colId.splice(index, 1);
                        if(colId.length === 0) delete appState[id.columnId];
                    }

                    saveState();
                    modal.classList.add('hidden');
                    window.currentDeleteId = null; // Reset global var
                    
                    fireConfetti();
                };
            };

            // Re-attach click listener to the DOM button explicitly in this init phase
            confirmBtn.onclick = () => {
                if(window.currentDeleteId) {
                    const id = window.currentDeleteId;
                    // Logic moved to the onclick handler above (inside the function) but simplified here:
                    
                    const colId = appState[id.columnId];
                    const index = colId.findIndex(c => c.id === id);
                    
                    if (index > -1) {
                        colId.splice(index, 1);
                        if(colId.length === 0) delete appState[id.columnId];
                    }

                    saveState();
                    modal.classList.add('hidden');
                    
                    fireConfetti();
                }
            };

        });

    </script>

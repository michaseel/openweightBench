        <!-- Columns container -->
        <div class="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6">
            <!-- Backlog Column -->

    <!-- Main JavaScript -->
    <script>
        // Kanban board application code
        document.addEventListener('DOMContentLoaded', () => {
            // Configuration and constants
            const STORAGE_NAMESPACE = 'kanban_board_state';
            
            // DOM elements
            const columns = {
                backlog: document.getElementById('backlog-list'),
                inProgress: document.getElementById('in-progress-list'),
                review: document.getElementById('review-list'),
                done: document.getElementById('done-list')
            };
            
            const columnCounts = {
                backlog: document.getElementById('backlog-count'),
                inProgress: document.getElementById('in-progress-count'),
                review: document.getElementById('review-count'),
                done: document.getElementById('done-count')
            };
            
            const deleteModal = document.getElementById('delete-modal');
            const confirmDeleteBtn = document.getElementById('confirm-delete');
            const cancelDeleteBtn = document.getElementById('cancel-delete');
            
            // State management
            let boardState = {
                backlog: [],
                inProgress: [],
                review: [],
                done: []
            };
            
            let cardToDelete = null;
            let draggedCard = null;
            
            // Initialize the board
            function initBoard() {
                // Load state from localStorage or create dummy cards if none exist
                const savedState = localStorage.getItem(STORAGE_NAMESPACE);
                
                if (savedState) {
                    boardState = JSON.parse(savedState);
                } else {
                    // Create dummy cards if none exist
                    boardState = {
                        backlog: [
                            { id: 'card-1', title: 'Design Homepage UI', description: 'Create wireframes and mockups for the homepage' },
                            { id: 'card-2', title: 'Implement Authentication', description: 'Set up user login and registration' },
                            { id: 'card-3', title: 'Database Schema', description: 'Design and implement the database structure' }
                        ],
                        inProgress: [
                            { id: 'card-4', title: 'API Integration', description: 'Connect frontend to backend services' }
                        ],
                        review: [],
                        done: []
                    };
                    
                    // Save the initial state
                    saveState();
                }
                
                renderBoard();
            }
            
            // Save the current state to localStorage
            function saveState() {
                localStorage.setItem(STORAGE_NAMESPACE, JSON.stringify(boardState));
            }
            
            // Render the entire board
            function renderBoard() {
                // Clear all columns first
                Object.values(columns).forEach(column => {
                    column.innerHTML = '';
                });
                
                // Render each card in its respective column
                Object.entries(boardState).forEach(([columnName, cards]) => {
                    const count = cards.length;
                    
                    // Update column count
                    if (columnCounts[columnName]) {
                        columnCounts[columnName].textContent = count;
                    }
                    
                    // Create and append each card
                    cards.forEach(card => {
                        const cardElement = createCardElement(card, columnName);
                        columns[columnName].appendChild(cardElement);
                    });
                });
            }
            
            // Create a card element
            function createCardElement(card, columnName) {
                const cardDiv = document.createElement('div');
                cardDiv.className = 'card bg-gray-50 p-3 rounded-lg shadow-sm cursor-move';
                cardDiv.draggable = true;
                cardDiv.dataset.cardId = card.id;
                cardDiv.setAttribute('data-testid', 'card');
                
                // Card content
                const cardContent = document.createElement('div');
                cardContent.className = 'mb-2';
                
                const titleDiv = document.createElement('div');
                titleDiv.className = 'font-medium text-gray-800 mb-1';
                titleDiv.textContent = card.title;
                
                const descriptionDiv = document.createElement('div');
                descriptionDiv.className = 'text-sm text-gray-600';
                descriptionDiv.textContent = card.description;
                
                // Edit button
                const editBtn = document.createElement('button');
                editBtn.className = 'text-xs text-blue-500 hover:text-blue-700 mt-1 mr-2';
                editBtn.textContent = 'Edit';
                
                // Delete button
                const deleteBtn = document.createElement('button');
                deleteBtn.className = 'text-xs text-red-500 hover:text-red-700 mt-1';
                deleteBtn.textContent = 'Delete';
                deleteBtn.setAttribute('data-testid', 'delete-card');
                
                cardContent.appendChild(titleDiv);
                cardContent.appendChild(descriptionDiv);
                cardDiv.appendChild(cardContent);
                
                const buttonContainer = document.createElement('div');
                buttonContainer.className = 'flex justify-end';
                
                editBtn.addEventListener('click', (e) => {
                    e.stopPropagation();
                    editCard(card, cardDiv);
                });
                
                deleteBtn.addEventListener('click', (e) => {
                    e.stopPropagation();
                    showDeleteConfirmation(card, cardDiv);
                });
                
                buttonContainer.appendChild(editBtn);
                buttonContainer.appendChild(deleteBtn);
                cardDiv.appendChild(buttonContainer);
                
                // Drag and drop events
                cardDiv.addEventListener('dragstart', handleDragStart);
                cardDiv.addEventListener('dragend', handleDragEnd);
                
                // Drop zone events for the card
                cardDiv.addEventListener('dragover', handleDragOver);
                cardDiv.addEventListener('drop', (e) => {
                    e.preventDefault();
                    handleDrop(e, columnName);
                });
                
                return cardDiv;
            }
            
            // Handle drag start
            function handleDragStart(e) {
                draggedCard = this;
                this.classList.add('dragging');
                
                // Store the card data in the drag event
                e.dataTransfer.effectAllowed = 'move';
                e.dataTransfer.setData('text/html', this.innerHTML);
            }
            
            // Handle drag end
            function handleDragEnd(e) {
                this.classList.remove('dragging');
                
                // Remove drag-over class from all columns
                Object.values(columns).forEach(column => {
                    column.classList.remove('drag-over');
                });
            }
            
            // Handle drag over a column
            function handleDragOver(e) {
                if (e.preventDefault) {
                    e.preventDefault();
                }
                
                e.dataTransfer.dropEffect = 'move';
                
                // Add visual feedback
                this.classList.add('drag-over');
                return false;
            }
            
            // Handle drop event
            function handleDrop(e, targetColumnName) {
                if (e.stopPropagation) {
                    e.stopPropagation();
                }
                
                // Get the source column name
                const cardId = draggedCard.dataset.cardId;
                let sourceColumnName = null;
                
                // Find which column the card came from
                for (const [name, cards] of Object.entries(boardState)) {
                    if (cards.some(card => card.id === cardId)) {
                        sourceColumnName = name;
                        break;
                    }
                }
                
                // Move the card if it's not already in the target column
                if (sourceColumnName && sourceColumnName !== targetColumnName) {
                    // Find and remove the card from the source column
                    const cardIndex = boardState[sourceColumnName].findIndex(card => card.id === cardId);
                    if (cardIndex !== -1) {
                        const [movedCard] = boardState[sourceColumnName].splice(cardIndex, 1);
                        
                        // Add to the target column
                        boardState[targetColumnName].push(movedCard);
                        
                        // Save state and re-render
                        saveState();
                        renderBoard();
                        
                        // Show confetti if moving to "Done" column
                        if (targetColumnName === 'done') {
                            showConfetti();
                        }
                    }
                }
                
                return false;
            }
            
            // Show confetti animation
            function showConfetti() {
                const container = document.getElementById('confetti-container');
                
                // Create confetti particles
                for (let i = 0; i < 100; i++) {
                    const confetti = document.createElement('div');
                    confetti.className = 'confetti';
                    
                    // Random position
                    confetti.style.left = `${Math.random() * 100}%`;
                    
                    // Random color
                    const colors = ['#3B82F6', '#8B5CF6', '#10B981', '#F59E0B', '#EF4444'];
                    confetti.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
                    
                    // Random size
                    const size = Math.random() * 10 + 5;
                    confetti.style.width = `${size}px`;
                    confetti.style.height = `${size}px`;
                    
                    // Random animation duration
                    const duration = Math.random() * 2 + 1;
                    confetti.style.animationDuration = `${duration}s`;
                    
                    container.appendChild(confetti);
                }
                
                // Remove confetti after animation
                setTimeout(() => {
                    container.innerHTML = '';
                }, 3000);
            }
            
            // Edit a card
            function editCard(card, cardElement) {
                const titleDiv = cardElement.querySelector('.font-medium');
                const descriptionDiv = cardElement.querySelector('.text-sm.text-gray-600');
                
                // Create input elements
                const titleInput = document.createElement('input');
                titleInput.type = 'text';
                titleInput.value = card.title;
                titleInput.className = 'font-medium text-gray-800 mb-1 w-full p-1 border rounded';
                
                const descriptionInput = document.createElement('textarea');
                descriptionInput.value = card.description;
                descriptionInput.className = 'text-sm text-gray-600 w-full p-1 border rounded';
                
                // Replace content with inputs
                titleDiv.replaceWith(titleInput);
                descriptionDiv.replaceWith(descriptionInput);
                
                // Focus on the title input
                titleInput.focus();
                
                // Handle saving the edited card
                function saveEdit() {
                    const newTitle = titleInput.value.trim();
                    const newDescription = descriptionInput.value.trim();
                    
                    if (newTitle) {
                        // Update the card in state
                        for (const [columnName, cards] of Object.entries(boardState)) {
                            const cardIndex = cards.findIndex(c => c.id === card.id);
                            if (cardIndex !== -1) {
                                cards[cardIndex].title = newTitle;
                                cards[cardIndex].description = newDescription;
                            }
                        }
                        
                        // Save state and re-render
                        saveState();
                        renderBoard();
                    } else {
                        // If title is empty, just re-render without changes
                        renderBoard();
                    }
                }
                
                // Save on Enter key press or blur
                titleInput.addEventListener('keydown', (e) => {
                    if (e.key === 'Enter') {
                        saveEdit();
                    }
                });
                
                descriptionInput.addEventListener('keydown', (e) => {
                    if (e.key === 'Enter' && !e.shiftKey) {
                        e.preventDefault();
                        saveEdit();
                    }
                });
                
                titleInput.addEventListener('blur', saveEdit);
                descriptionInput.addEventListener('blur', saveEdit);
            }
            
            // Show delete confirmation modal
            function showDeleteConfirmation(card, cardElement) {
                cardToDelete = { card, element: cardElement };
                
                // Show the modal
                deleteModal.classList.remove('hidden');
            }
            
            // Hide delete confirmation modal
            function hideDeleteConfirmation() {
                deleteModal.classList.add('hidden');
                cardToDelete = null;
            }
            
            // Delete a card after confirmation
            function deleteCard() {
                if (cardToDelete) {
                    const { card, element } = cardToDelete;
                    
                    // Remove from state
                    for (const [columnName, cards] of Object.entries(boardState)) {
                        const cardIndex = cards.findIndex(c => c.id === card.id);
                        if (cardIndex !== -1) {
                            cards.splice(cardIndex, 1);
                        }
                    }
                    
                    // Save state and re-render
                    saveState();
                    renderBoard();
                }
                
                hideDeleteConfirmation();
            }
            
            // Add a new card to a column
            function addCard(columnName) {
                const input = document.createElement('input');
                input.type = 'text';
                input.placeholder = 'Enter card title...';
                input.className = 'w-full p-2 border rounded mb-1';
                input.setAttribute('data-testid', 'card-input');
                
                const descriptionInput = document.createElement('textarea');
                descriptionInput.placeholder = 'Enter card description (optional)...';
                descriptionInput.className = 'w-full p-2 border rounded mb-1';
                
                const buttonContainer = document.createElement('div');
                buttonContainer.className = 'flex justify-between';
                
                const addBtn = document.createElement('button');
                addBtn.textContent = 'Add';
                addBtn.className = 'px-3 py-1 bg-blue-500 text-white rounded hover:bg-blue-600';
                
                const cancelBtn = document.createElement('button');
                cancelBtn.textContent = 'Cancel';
                cancelBtn.className = 'px-3 py-1 bg-gray-300 text-gray-700 rounded hover:bg-gray-400';
                
                buttonContainer.appendChild(addBtn);
                buttonContainer.appendChild(cancelBtn);
                
                const cardDiv = document.createElement('div');
                cardDiv.className = 'bg-gray-50 p-3 rounded-lg shadow-sm';
                
                cardDiv.appendChild(input);
                cardDiv.appendChild(descriptionInput);
                cardDiv.appendChild(buttonContainer);
                
                // Insert at the beginning of the column
                columns[columnName].insertBefore(cardDiv, columns[columnName].firstChild);
                
                // Focus on the input
                input.focus();
                
                // Handle adding the card
                function handleAdd() {
                    const title = input.value.trim();
                    
                    if (title) {
                        // Create a new card
                        const newCard = {
                            id: `card-${Date.now()}`,
                            title,
                            description: descriptionInput.value.trim()
                        };
                        
                        // Add to the beginning of the column
                        boardState[columnName].unshift(newCard);
                        
                        // Save state and re-render
                        saveState();
                        renderBoard();
                    } else {
                        // If title is empty, just re-render without changes
                        renderBoard();
                    }
                }
                
                // Add on Enter key press or button click
                input.addEventListener('keydown', (e) => {
                    if (e.key === 'Enter') {
                        handleAdd();
                    }
                });
                
                addBtn.addEventListener('click', handleAdd);
                
                // Cancel button
                cancelBtn.addEventListener('click', () => {
                    renderBoard();
                });
            }
            
            // Event listeners for "Add card" buttons
            document.querySelectorAll('[data-testid="add-card"]').forEach(button => {
                button.addEventListener('click', (e) => {
                    // Find the column name based on which button was clicked
                    const columnDiv = e.target.closest('[data-testid*="column-"]');
                    
                    if (columnDiv) {
                        const columnTestId = columnDiv.dataset.testid;
                        
                        // Map test IDs to internal column names
                        const testNameToColumn = {
                            'column-backlog': 'backlog',
                            'column-in-progress': 'inProgress',
                            'column-review': 'review',
                            'column-done': 'done'
                        };
                        
                        const columnName = testNameToColumn[columnTestId];
                        
                        if (columnName) {
                            addCard(columnName);
                        }
                    }
                });
            });
            
            // Event listeners for delete confirmation modal
            confirmDeleteBtn.addEventListener('click', deleteCard);
            cancelDeleteBtn.addEventListener('click', hideDeleteConfirmation);
            
            // Close modal when clicking outside
            deleteModal.addEventListener('click', (e) => {
                if (e.target === deleteModal) {
                    hideDeleteConfirmation();
                }
            });
            
            // Drop zone events for columns
            Object.entries(columns).forEach(([name, column]) => {
                column.addEventListener('dragover', handleDragOver);
                column.addEventListener('drop', (e) => {
                    e.preventDefault();
                    handleDrop(e, name);
                });
                
                // Remove drag-over class when dragging leaves
                column.addEventListener('dragleave', (e) => {
                    if (!column.contains(e.relatedTarget)) {
                        column.classList.remove('drag-over');
                    }
                });
            });
            
            // Initialize the board when DOM is ready
            initBoard();
        });
    </script>
</body>
